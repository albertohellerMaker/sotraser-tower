CAMBIO 2 — Umbral adaptativo por tipo de geocerca

Crear función getUmbralParada(lugar: string | null):

function getUmbralParada(
  lat: number | null,
  lng: number | null,
  geocercas: GeoBase[]
): number {
  
  // Si no hay coordenadas → usar 90 min
  if (!lat || !lng) return 90
  
  // Buscar geocerca más cercana (radio 1km)
  const cercana = geocercas.find(g => {
    const dist = haversineKm(
      lat, lng,
      parseFloat(g.lat),
      parseFloat(g.lng)
    )
    return dist < 1.0
  })
  
  if (!cercana) return 90 // Sin geocerca → 90 min
  
  // Umbral según tipo de lugar
  switch (cercana.tipo) {
    case 'BASE':
      return 480  // 8 horas — puede pernoctar
    case 'MINA':
      return 360  // 6 horas — espera de turno
    case 'CD':
      return 180  // 3 horas — descarga compleja
    case 'ESTACION':
      return 45   // 45 min — solo cargar combustible
    case 'PATIO':
      return 240  // 4 horas — operación de patio
    default:
      return 90   // 90 min — caso general
  }
}
