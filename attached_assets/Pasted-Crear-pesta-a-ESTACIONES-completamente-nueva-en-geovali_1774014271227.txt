async function getAnalisisEstaciones(req, res) {
  const ahora = new Date()
  const hace24h = new Date(ahora.getTime() - 24*60*60*1000)
  
  // ─── 1. CARGAS SIGETRA (últimas 24h) ──────────────
  const cargasSigetra = await db.select()
    .from(cargas)
    .where(gte(cargas.fecha, hace24h))
    .orderBy(desc(cargas.fecha))
  
  // ─── 2. SNAPSHOTS VOLVO (últimas 24h) ─────────────
  // Para cruzar nivel de estanque pre/post carga
  const snapshots = await db.select()
    .from(volvoFuelSnapshots)
    .where(gte(volvoFuelSnapshots.capturedAt, hace24h))
    .orderBy(asc(volvoFuelSnapshots.capturedAt))
  
  // ─── 3. GEOCERCAS DE TIPO ESTACION ────────────────
  const geocercasEstacion = await db.select()
    .from(geoBases)
    .where(
      and(
        eq(geoBases.tipo, 'ESTACION'),
        eq(geoBases.activo, true)
      )
    )
  
  // ─── 4. CAMIONES CON VIN ──────────────────────────
  const camionesConVin = await db.select()
    .from(camiones)
    .where(isNotNull(camiones.vin))
  
  const vinPorPatente: Record<string, string> = {}
  for (const c of camionesConVin) {
    if (c.patente && c.vin) vinPorPatente[c.patente] = c.vin
  }
  
  // ─── 5. ANÁLISIS POR ESTACIÓN ─────────────────────
  
  const porEstacion: Record<string, {
    nombre: string
    ciudad: string
    lat: number | null
    lng: number | null
    cargas: CargaAnomalia[]
    total_litros: number
    total_cargas: number
    camiones_distintos: number
    tiene_anomalias: boolean
    alertas_count: number
  }> = {}
  
  for (const carga of cargasSigetra) {
    const lugar = carga.lugar || 'Desconocido'
    
    if (!porEstacion[lugar]) {
      // Buscar coordenadas en geocercas
      const geo = geocercasEstacion.find(g => 
        g.nombre.toLowerCase().includes(lugar.toLowerCase()) ||
        lugar.toLowerCase().includes(g.nombre.toLowerCase())
      )
      
      porEstacion[lugar] = {
        nombre: lugar,
        ciudad: carga.ciudad || '',
        lat: geo ? parseFloat(geo.lat) : null,
        lng: geo ? parseFloat(geo.lng) : null,
        cargas: [],
        total_litros: 0,
        total_cargas: 0,
        camiones_distintos: 0,
        tiene_anomalias: false,
        alertas_count: 0
      }
    }
    
    // Analizar cada carga individualmente
    const anomalia = await analizarCarga(
      carga, 
      snapshots, 
      vinPorPatente
    )
    
    porEstacion[lugar].cargas.push(anomalia)
    porEstacion[lugar].total_litros += carga.litros || 0
    porEstacion[lugar].total_cargas++
    
    if (anomalia.nivel_alerta !== 'NORMAL') {
      porEstacion[lugar].tiene_anomalias = true
      porEstacion[lugar].alertas_count++
    }
  }
  
  // Calcular camiones distintos por estación
  for (const est of Object.values(porEstacion)) {
    const patentes = new Set(est.cargas.map(c => c.patente))
    est.camiones_distintos = patentes.size
  }
  
  // Ordenar: primero con anomalías, luego por litros
  const estaciones = Object.values(porEstacion)
    .sort((a, b) => {
      if (a.alertas_count !== b.alertas_count) 
        return b.alertas_count - a.alertas_count
      return b.total_litros - a.total_litros
    })
  
  // ─── 6. RESUMEN GLOBAL ────────────────────────────
  const resumen = {
    total_estaciones: estaciones.length,
    total_cargas: cargasSigetra.length,
    total_litros: cargasSigetra.reduce((s,c) => s + (c.litros||0), 0),
    estaciones_con_anomalias: estaciones.filter(e => e.tiene_anomalias).length,
    cargas_anomalas: estaciones.reduce((s,e) => s + e.alertas_count, 0),
    cargas_criticas: estaciones.reduce((s,e) => 
      s + e.cargas.filter(c => c.nivel_alerta === 'CRITICO').length, 0),
    periodo: { desde: hace24h, hasta: ahora }
  }
  
  res.json({ resumen, estaciones })
}

// ─── FUNCIÓN CORE: ANALIZAR UNA CARGA ─────────────────

async function analizarCarga(
  carga: any,
  snapshots: any[],
  vinPorPatente: Record<string, string>
): Promise<CargaAnomalia> {
  
  const vin = vinPorPatente[carga.patente]
  const fechaCarga = new Date(carga.fecha)
  
  // Ventana de análisis: 30 min antes y 45 min después
  const ventanaAntes = new Date(fechaCarga.getTime() - 30*60*1000)
  const ventanaDespues = new Date(fechaCarga.getTime() + 45*60*1000)
  
  let nivelAlerta: 'NORMAL' | 'REVISAR' | 'SOSPECHOSO' | 'CRITICO' = 'NORMAL'
  const razones: string[] = []
  let litrosDelta: number | null = null
  let litrosEcuIncrementoReal: number | null = null
  
  // ─── SEÑAL 1: CRUCE ECU vs SIGETRA ────────────────
  if (vin) {
    const snapsAntes = snapshots.filter(s => 
      s.vin === vin && 
      new Date(s.capturedAt) >= ventanaAntes &&
      new Date(s.capturedAt) <= fechaCarga
    ).sort((a,b) => 
      new Date(b.capturedAt).getTime() - new Date(a.capturedAt).getTime()
    )
    
    const snapsDespues = snapshots.filter(s =>
      s.vin === vin &&
      new Date(s.capturedAt) > fechaCarga &&
      new Date(s.capturedAt) <= ventanaDespues
    ).sort((a,b) =>
      new Date(a.capturedAt).getTime() - new Date(b.capturedAt).getTime()
    )
    
    if (snapsAntes.length > 0 && snapsDespues.length > 0) {
      const fuelAntes = snapsAntes[0].totalFuelUsed
      const fuelDespues = snapsDespues[0].totalFuelUsed
      
      // totalFuelUsed es acumulativo — si baja, se cargó
      // La diferencia negativa = litros cargados según ECU
      const deltaECU = fuelAntes - fuelDespues
      litrosEcuIncrementoReal = deltaECU > 0 ? deltaECU / 1000 : null
      
      if (litrosEcuIncrementoReal !== null && carga.litros > 0) {
        litrosDelta = carga.litros - litrosEcuIncrementoReal
        const pctDiferencia = Math.abs(litrosDelta) / carga.litros * 100
        
        // Si Sigetra dice más litros que lo que entró al estanque
        if (litrosDelta > 50 && pctDiferencia > 20) {
          nivelAlerta = 'CRITICO'
          razones.push(
            `Sigetra: ${carga.litros}L declarados, ` +
            `ECU: ${litrosEcuIncrementoReal.toFixed(0)}L ingresados real. ` +
            `Diferencia: ${litrosDelta.toFixed(0)}L no entraron al estanque`
          )
        } else if (litrosDelta > 25 && pctDiferencia > 10) {
          nivelAlerta = 'SOSPECHOSO'
          razones.push(
            `Diferencia ${litrosDelta.toFixed(0)}L entre ` +
            `Sigetra (${carga.litros}L) y ECU (${litrosEcuIncrementoReal.toFixed(0)}L)`
          )
        }
      }
    }
  }
  
  // ─── SEÑAL 2: CARGA EXCESIVA ──────────────────────
  // Si carga más del 120% de la capacidad típica del camión
  if (carga.litros > 900) {
    nivelAlerta = nivelAlerta === 'NORMAL' ? 'SOSPECHOSO' : nivelAlerta
    razones.push(`Volumen inusualmente alto: ${carga.litros}L en una sola carga`)
  }
  
  // ─── SEÑAL 3: HORARIO INUSUAL ─────────────────────
  const hora = fechaCarga.getHours()
  const esMadrugada = hora >= 1 && hora <= 4
  
  if (esMadrugada) {
    nivelAlerta = nivelAlerta === 'NORMAL' ? 'REVISAR' : nivelAlerta
    razones.push(`Carga en horario inusual: ${hora}:${String(fechaCarga.getMinutes()).padStart(2,'0')} AM`)
  }
  
  // ─── SEÑAL 4: ESTACIÓN NO HABITUAL ───────────────
  // Verificar contra historial de esta patente en esta ruta
  // (simplificado por ahora — se enriquece con el aprendizaje)
  const esEstacionConocida = carga.lugar && (
    carga.lugar.toLowerCase().includes('copec') ||
    carga.lugar.toLowerCase().includes('shell') ||
    carga.lugar.toLowerCase().includes('servicentro')
  )
  
  if (!esEstacionConocida && carga.lugar) {
    nivelAlerta = nivelAlerta === 'NORMAL' ? 'REVISAR' : nivelAlerta
    razones.push(`Estación no identificada como punto autorizado: "${carga.lugar}"`)
  }
  
  // ─── SEÑAL 5: MÚLTIPLES CARGAS MISMO DÍA ─────────
  // Contar cuántas cargas tiene esta patente hoy
  // (se calcula fuera de esta función con el contexto completo)
  
  return {
    id: carga.id,
    patente: carga.patente || 'Sin patente',
    conductor: carga.conductor || 'Sin asignar',
    contrato: carga.contrato || '',
    fecha: carga.fecha,
    hora: fechaCarga.toLocaleTimeString('es-CL', { hour: '2-digit', minute: '2-digit' }),
    litros_sigetra: carga.litros || 0,
    litros_ecu_real: litrosEcuIncrementoReal,
    litros_delta: litrosDelta,
    nivel_alerta: nivelAlerta,
    razones,
    tiene_cruce_ecu: litrosEcuIncrementoReal !== null,
    odometro: carga.odometro || null
  }
}

  if (!data) return null

function EstacionRow({ estacion, abierta, onToggle }) {
  
  const maxAlerta = estacion.cargas.reduce((max, c) => {
    const orden = { NORMAL: 0, REVISAR: 1, SOSPECHOSO: 2, CRITICO: 3 }
    return orden[c.nivel_alerta] > orden[max] ? c.nivel_alerta : max
  }, 'NORMAL')
  
  const colorAlerta = {
    NORMAL:     '#00ff88',
    REVISAR:    '#00d4ff',
    SOSPECHOSO: '#ff8c00',
    CRITICO:    '#ff2244'
  }[maxAlerta]
  
  const iconAlerta = {
    NORMAL:     '✅',
    REVISAR:    '🔵',
    SOSPECHOSO: '🟡',
    CRITICO:    '🔴'
  }[maxAlerta]

  return (
    <div style={{ border: `1px solid ${maxAlerta !== 'NORMAL' ? colorAlerta + '40' : '#0d2035'}` }}>
      
      {/* ── FILA PRINCIPAL ────────────── */}
      <button
        onClick={onToggle}
        className="w-full p-3 flex items-center gap-3 cursor-pointer
                   transition-all hover:bg-[rgba(0,212,255,0.02)]"
        style={{ background: maxAlerta !== 'NORMAL' ? `${colorAlerta}08` : '#060d14' }}
      >
        {/* Ícono estado */}
        <span className="text-sm flex-shrink-0">{iconAlerta}</span>
        
        {/* Nombre + ciudad */}
        <div className="flex-1 min-w-0 text-left">
          <div className="font-space text-[11px] font-bold truncate"
            style={{ color: '#c8e8ff' }}>
            {estacion.nombre}
          </div>
          {estacion.ciudad && (
            <div className="font-exo text-[9px]" style={{ color: '#3a6080' }}>
              {estacion.ciudad}
            </div>
          )}
        </div>
        
        {/* Barra de actividad */}
        <div className="hidden md:block w-24">
          <div className="h-1 w-full" style={{ background: '#0d2035' }}>
            <div className="h-full transition-all"
              style={{ 
                width: `${Math.min(estacion.total_litros / 2000 * 100, 100)}%`,
                background: colorAlerta

        </div>
        
        {/* Métricas */}

          <div className="text-right">
            <div className="font-space text-[12px] font-bold"
              style={{ color: '#ffcc00' }}>
              {estacion.total_litros.toLocaleString('es-CL')}L
            </div>
            <div className="font-exo text-[8px]" style={{ color: '#3a6080' }}>
              litros
            </div>
          </div>
          <div className="text-right">
            <div className="font-space text-[12px] font-bold"
              style={{ color: '#c8e8ff' }}>
              {estacion.total_cargas}
            </div>
            <div className="font-exo text-[8px]" style={{ color: '#3a6080' }}>
              cargas
            </div>
          </div>
          <div className="text-right">
            <div className="font-space text-[12px] font-bold"
              style={{ color: '#c8e8ff' }}>
              {estacion.camiones_distintos}
            </div>
            <div className="font-exo text-[8px]" style={{ color: '#3a6080' }}>
              camiones
            </div>
          </div>
          
          {/* Badge anomalías */}
          {estacion.alertas_count > 0 && (
            <div className="px-2 py-1 font-space text-[9px] font-bold"
              style={{

                color: colorAlerta

              {estacion.alertas_count} alerta{estacion.alertas_count > 1 ? 's' : ''}

          )}
          
          {/* Chevron */}
          <span style={{ color: '#3a6080' }}>

      </button>

      {/* ── DETALLE EXPANDIDO ─────────── */}
      {abierta && (

          <div className="p-3">

              style={{ color: '#3a6080' }}>

function CargaFila({ carga }) {
  const [expandida, setExpandida] = useState(false)
  
  const colores = {
    NORMAL:     '#00ff88',
    REVISAR:    '#00d4ff',
    SOSPECHOSO: '#ff8c00',
    CRITICO:    '#ff2244'
  }
  const color = colores[carga.nivel_alerta]
  
  return (
    <div>
      <button
        onClick={() => setExpandida(!expandida)}
        className="w-full grid grid-cols-7 px-2 py-2 cursor-pointer
                   transition-all hover:bg-[rgba(255,255,255,0.02)]"
        style={{ 
          background: carga.nivel_alerta !== 'NORMAL' ? `${color}05` : 'transparent',
          borderLeft: carga.nivel_alerta !== 'NORMAL' ? `2px solid ${color}` : '2px solid transparent'
        }}
      >
        {/* Hora */}
        <div className="font-space text-[10px]" style={{ color: '#c8e8ff' }}>
          {carga.hora}
        </div>
        
        {/* Patente */}
        <div className="font-space text-[10px] font-bold" style={{ color: '#00d4ff' }}>
          {carga.patente}
        </div>
        
        {/* Conductor */}
        <div className="font-exo text-[10px] truncate" style={{ color: '#3a6080' }}>
          {carga.conductor}
        </div>
        
        {/* Litros Sigetra */}
        <div className="font-space text-[10px] font-bold" style={{ color: '#ffcc00' }}>
          {carga.litros_sigetra}L
        </div>
        
        {/* Litros ECU real */}
        <div className="font-space text-[10px]" 
          style={{ color: carga.tiene_cruce_ecu ? '#c8e8ff' : '#3a6080' }}>
          {carga.litros_ecu_real !== null 
            ? `${Math.round(carga.litros_ecu_real)}L`
            : '—'}
        </div>
        
        {/* Delta */}
        <div className="font-space text-[10px] font-bold"
          style={{ 
            color: carga.litros_delta === null ? '#3a6080' :
                   carga.litros_delta > 25 ? '#ff2244' :
                   carga.litros_delta > 10 ? '#ff8c00' : '#00ff88'
          }}>
          {carga.litros_delta !== null 
            ? `${carga.litros_delta > 0 ? '+' : ''}${Math.round(carga.litros_delta)}L`
            : '—'}
        </div>
        
        {/* Estado badge */}
        <div>
          <span className="font-exo text-[8px] font-bold px-1.5 py-0.5"
            style={{ 
              color, 
              background: `${color}15`,
              border: `1px solid ${color}40`
            }}>
            {carga.nivel_alerta}
          </span>
        </div>
      </button>
      
      {/* Razones expandidas */}
      {expandida && carga.razones.length > 0 && (
        <div className="px-4 py-2 space-y-1"
          style={{ background: '#0a1520', borderTop: '1px solid #0d2035' }}>
          <div className="font-exo text-[8px] tracking-wider uppercase mb-1"
            style={{ color: '#3a6080' }}>
            RAZONES DE ALERTA:
          </div>
          {carga.razones.map((r, i) => (
            <div key={i} className="font-rajdhani text-[11px] flex items-start gap-2"
              style={{ color: '#c8e8ff' }}>
              <span style={{ color: '#ff2244' }}>→</span>
              {r}
            </div>
          ))}
          
          {/* Botón consultar IA */}
          <button
            onClick={async () => {
              // Abrir chat con contexto pre-cargado
              const msg = `Analiza esta carga de combustible sospechosa: 
                Patente ${carga.patente}, conductor ${carga.conductor}, 
                ${carga.hora} en ${carga.estacion}. 
                Sigetra declaró ${carga.litros_sigetra}L pero 
                ECU registró ${carga.litros_ecu_real}L.
                Diferencia: ${carga.litros_delta}L. 
                ¿Qué puede haber ocurrido?`
              // Disparar evento al chat flotante
              window.dispatchEvent(new CustomEvent('sotra-ia-consulta', { 
                detail: { mensaje: msg }
              }))
            }}
            className="mt-1 font-exo text-[9px] px-3 py-1 border cursor-pointer
                       transition-all hover:border-[#00d4ff]"
            style={{ borderColor: '#0d2035', color: '#3a6080' }}
          >
            🧠 Consultar SOTRA IA sobre esta carga
          </button>
        </div>
      )}
    </div>
  )
}

// Detectar camiones con múltiples cargas en el día
const cargasPorPatente: Record<string, number> = {}

for (const est of Object.values(porEstacion)) {
  for (const carga of est.cargas) {
    cargasPorPatente[carga.patente] = 
      (cargasPorPatente[carga.patente] || 0) + 1
  }
}

// Marcar cargas de patentes con 3+ cargas en el día
for (const est of Object.values(porEstacion)) {
  for (const carga of est.cargas) {
    const totalCargas = cargasPorPatente[carga.patente] || 0
    if (totalCargas >= 3) {
      if (carga.nivel_alerta === 'NORMAL') {
        carga.nivel_alerta = 'REVISAR'
      }
      carga.razones.push(
        `Camión con ${totalCargas} cargas en las últimas 24h`
      )
      est.alertas_count++
      est.tiene_anomalias = true
    }
  }
}
