app.get("/api/contratos/cuadratura-mensual",
async (req, res) => {
  const { contrato } = req.query
  
  // Período: desde día 1 del mes hasta T-1
  const hoy = new Date()
  const ayer = new Date(hoy)
  ayer.setDate(ayer.getDate() - 1)
  ayer.setHours(23, 59, 59, 999)
  
  const inicioMes = new Date(hoy.getFullYear(), hoy.getMonth(), 1)
  inicioMes.setHours(0, 0, 0, 0)
  
  // Solo operaciones CERRADAS y VERIFICADAS
  // PERFECTA + BUENA + ACEPTABLE
  // NO incluir PENDIENTE ni CON_ALERTA en cuadratura
  
  let whereClause = and(
    gte(operacionesExitosas.fechaSalida, inicioMes),
    lte(operacionesExitosas.fechaSalida, ayer),
    inArray(operacionesExitosas.calidadCierre, [
      'PERFECTA', 'BUENA', 'ACEPTABLE'
    ])
  )
  
  if (contrato && contrato !== 'TODOS') {
    whereClause = and(
      whereClause,
      eq(operacionesExitosas.contrato, contrato as string)
    )
  }
  
  const operaciones = await db.select()
    .from(operacionesExitosas)
    .where(whereClause)
    .orderBy(desc(operacionesExitosas.fechaSalida))
  
  // Agrupar por camión
  const porCamion: Record<string, {
    patente: string
    conductor: string
    contrato: string
    operaciones: typeof operaciones
    km_total: number
    litros_ecu_total: number
    litros_sigetra_total: number
    rendimiento_promedio: number | null
    operaciones_perfectas: number
    operaciones_buenas: number
    operaciones_aceptables: number
    operaciones_con_alerta: number
    balance_total: number
    mejor_rendimiento: number | null
    peor_rendimiento: number | null
    tendencia: 'MEJORANDO' | 'ESTABLE' | 'EMPEORANDO'
  }> = {}
  
  for (const op of operaciones) {
    const key = op.patente
    if (!porCamion[key]) {
      porCamion[key] = {
        patente: op.patente,
        conductor: op.conductor || 'Sin asignar',
        contrato: op.contrato || '',
        operaciones: [],
        km_total: 0,
        litros_ecu_total: 0,
        litros_sigetra_total: 0,
        rendimiento_promedio: null,
        operaciones_perfectas: 0,
        operaciones_buenas: 0,
        operaciones_aceptables: 0,
        operaciones_con_alerta: 0,
        balance_total: 0,
        mejor_rendimiento: null,
        peor_rendimiento: null,
        tendencia: 'ESTABLE'
      }
    }
    
    const c = porCamion[key]
    c.operaciones.push(op)
    c.km_total += op.kmEcu || 0
    c.litros_ecu_total += op.litrosEcu || 0
    c.litros_sigetra_total += op.litrosSigetra || 0
    
    if (op.calidadCierre === 'PERFECTA') c.operaciones_perfectas++
    else if (op.calidadCierre === 'BUENA') c.operaciones_buenas++
    else if (op.calidadCierre === 'ACEPTABLE') c.operaciones_aceptables++
    
    if (op.rendimientoReal) {
      if (!c.mejor_rendimiento || op.rendimientoReal > c.mejor_rendimiento)
        c.mejor_rendimiento = op.rendimientoReal
      if (!c.peor_rendimiento || op.rendimientoReal < c.peor_rendimiento)
        c.peor_rendimiento = op.rendimientoReal
    }
  }
  
  // Calcular promedios y tendencia
  for (const cam of Object.values(porCamion)) {
    if (cam.litros_ecu_total > 0 && cam.km_total > 0) {
      cam.rendimiento_promedio = Math.round(
        (cam.km_total / cam.litros_ecu_total) * 100
      ) / 100
    }
    
    cam.balance_total = Math.round(
      cam.litros_sigetra_total - cam.litros_ecu_total
    )
    
    // Tendencia: comparar primera mitad vs segunda mitad del mes
    const mitad = Math.floor(cam.operaciones.length / 2)
    if (mitad >= 2) {
      const primera = cam.operaciones.slice(0, mitad)
      const segunda = cam.operaciones.slice(mitad)
      
      const rendPrimera = primera
        .filter(o => o.rendimientoReal)
        .reduce((s,o) => s + o.rendimientoReal!, 0) / primera.length
      const rendSegunda = segunda
        .filter(o => o.rendimientoReal)
        .reduce((s,o) => s + o.rendimientoReal!, 0) / segunda.length
      
      const diff = ((rendSegunda - rendPrimera) / rendPrimera) * 100
      
      cam.tendencia = diff > 5 ? 'MEJORANDO'
                    : diff < -5 ? 'EMPEORANDO'
                    : 'ESTABLE'
    }
  }
  
  // Resumen global del contrato
  const camionesArr = Object.values(porCamion)
  
  const resumen = {
    periodo: {
      desde: inicioMes.toISOString().slice(0, 10),
      hasta: ayer.toISOString().slice(0, 10),
      dias: Math.floor(
        (ayer.getTime() - inicioMes.getTime()) 
        / (1000 * 60 * 60 * 24)
      )
    },
    total_operaciones: operaciones.length,
    camiones_activos: camionesArr.length,
    km_total: Math.round(
      camionesArr.reduce((s,c) => s + c.km_total, 0)
    ),
    litros_ecu_total: Math.round(
      camionesArr.reduce((s,c) => s + c.litros_ecu_total, 0)
    ),
    litros_sigetra_total: Math.round(
      camionesArr.reduce((s,c) => s + c.litros_sigetra_total, 0)
    ),
    balance_total: Math.round(
      camionesArr.reduce((s,c) => s + c.balance_total, 0)
    ),
    rendimiento_promedio: (() => {
      const totalKm = camionesArr.reduce((s,c) => s + c.km_total, 0)
      const totalL = camionesArr.reduce((s,c) => s + c.litros_ecu_total, 0)
      return totalL > 0 ? Math.round(totalKm / totalL * 100) / 100 : null
    })(),
    calidad_distribucion: {
      perfectas: operaciones.filter(o => o.calidadCierre === 'PERFECTA').length,
      buenas: operaciones.filter(o => o.calidadCierre === 'BUENA').length,
      aceptables: operaciones.filter(o => o.calidadCierre === 'ACEPTABLE').length,
    },
    cobertura_pct: Math.round(
      (operaciones.length / Math.max(1, camionesArr.length * 20)) * 100
    )
  }
  
  // Ordenar camiones: mayor balance primero (más sospechosos)
  const camionesOrdenados = camionesArr.sort((a, b) => 
    b.balance_total - a.balance_total
  )
  
  res.json({ resumen, camiones: camionesOrdenados })
})

app.get("/api/contratos/cuadratura-camion",
async (req, res) => {
  const { patente } = req.query
  if (!patente) return res.status(400).json({ error: 'patente requerida' })
  
  const hoy = new Date()
  const ayer = new Date(hoy)
  ayer.setDate(ayer.getDate() - 1)
  ayer.setHours(23, 59, 59, 999)
  const inicioMes = new Date(hoy.getFullYear(), hoy.getMonth(), 1)
  
  // Todas las operaciones del mes de ese camión
  const operaciones = await db.select()
    .from(operacionesExitosas)
    .where(
      and(
        eq(operacionesExitosas.patente, patente as string),
        gte(operacionesExitosas.fechaSalida, inicioMes),
        lte(operacionesExitosas.fechaSalida, ayer)
      )
    )
    .orderBy(desc(operacionesExitosas.fechaSalida))
  
  // Datos del camión
  const camionData = await db.select()
    .from(camiones)
    .where(eq(camiones.patente, patente as string))
    .limit(1)
  
  // Perfil aprendido
  const perfil = await db.select()
    .from(camionesPerfil)
    .where(eq(camionesPerfil.patente, patente as string))
    .limit(1)
  
  // Calcular métricas del mes
  const cerradas = operaciones.filter(o => 
    ['PERFECTA','BUENA','ACEPTABLE'].includes(o.calidadCierre)
  )
  const conAlerta = operaciones.filter(o => 
    o.calidadCierre === 'CON_ALERTA'
  )
  
  const kmTotal = cerradas.reduce((s,o) => s + (o.kmEcu || 0), 0)
  const litrosEcuTotal = cerradas.reduce((s,o) => s + (o.litrosEcu || 0), 0)
  const litrosSigetraTotal = cerradas.reduce((s,o) => s + (o.litrosSigetra || 0), 0)
  
  // Rendimiento por semana para el gráfico
  const porSemana: Record<string, { km: number, litros: number, ops: number }> = {}
  for (const op of cerradas) {
    const semana = getSemanaLabel(new Date(op.fechaSalida))
    if (!porSemana[semana]) porSemana[semana] = { km: 0, litros: 0, ops: 0 }
    porSemana[semana].km += op.kmEcu || 0
    porSemana[semana].litros += op.litrosEcu || 0
    porSemana[semana].ops++
  }
  
  const tendenciaSemanal = Object.entries(porSemana).map(([semana, d]) => ({
    semana,
    rendimiento: d.litros > 0 ? Math.round(d.km / d.litros * 100) / 100 : null,
    km: Math.round(d.km),
    litros: Math.round(d.litros),
    operaciones: d.ops
  }))
  
  res.json({
    patente,
    conductor: operaciones[0]?.conductor || 'Sin asignar',
    contrato: operaciones[0]?.contrato || '',
    modelo: camionData[0]?.modelo || '',
    meta_kmL: camionData[0]?.metaKmL || 2.85,
    
    resumen_mes: {
      total_operaciones: operaciones.length,
      operaciones_verificadas: cerradas.length,
      operaciones_con_alerta: conAlerta.length,
      km_total: Math.round(kmTotal),
      litros_ecu_total: Math.round(litrosEcuTotal),
      litros_sigetra_total: Math.round(litrosSigetraTotal),
      balance_litros: Math.round(litrosSigetraTotal - litrosEcuTotal),
      balance_pct: litrosEcuTotal > 0 
        ? Math.round((litrosSigetraTotal - litrosEcuTotal) / litrosEcuTotal * 100)
        : null,
      rendimiento_promedio: litrosEcuTotal > 0
        ? Math.round(kmTotal / litrosEcuTotal * 100) / 100
        : null,
      vs_meta: camionData[0]?.metaKmL && litrosEcuTotal > 0
        ? Math.round(
            ((kmTotal/litrosEcuTotal) - camionData[0].metaKmL) 
            / camionData[0].metaKmL * 100
          )
        : null
    },
    
    tendencia_semanal: tendenciaSemanal,
    
    perfil_aprendido: perfil[0] ? {
      total_jornadas: perfil[0].totalJornadas,
      score: perfil[0].scoreRendimiento,
      tendencia: perfil[0].tendencia,
      confianza: perfil[0].confianza || 'BAJA'
    } : null,
    
    operaciones: operaciones.map(op => ({
      id: op.id,
      fecha: op.fechaSalida,
      origen: op.origenNombre,
      destino: op.destinoNombre,
      km_ecu: op.kmEcu,
      litros_ecu: op.litrosEcu,
      litros_sigetra: op.litrosSigetra,
      rendimiento: op.rendimientoReal,
      duracion_horas: op.duracionHoras,
      calidad: op.calidadCierre,
      balance: op.litrosSigetra && op.litrosEcu
        ? Math.round(op.litrosSigetra - op.litrosEcu)
        : null
    }))
  })
})

function getSemanaLabel(fecha: Date): string {
  const inicioSemana = new Date(fecha)
  inicioSemana.setDate(fecha.getDate() - fecha.getDay())
  return `S${Math.ceil(fecha.getDate() / 7)}`
}

function MiniCuadroOperacional({
  patente,
  onClose
}: {
  patente: string
  onClose: () => void
}) {
  
  const { data, isLoading } = useQuery({
    queryKey: ['/api/contratos/cuadratura-camion', patente],
    queryFn: () => fetch(
      `/api/contratos/cuadratura-camion?patente=` +
      encodeURIComponent(patente)
    ).then(r => r.json())
  })
  
  return (
    <div className="fixed inset-0 z-[60] flex items-center 
                    justify-center"
      onClick={onClose}>
      <div className="absolute inset-0"
        style={{ background: 'rgba(2,5,8,0.7)' }} />
      
      <div className="relative w-full max-w-[640px]

                      const color = op.calidad === 'PERFECTA'
                        ? '#00ff88'
                        : op.calidad === 'BUENA' ? '#00d4ff'
                        : op.calidad === 'ACEPTABLE' ? '#ffcc00'
                        : '#ff2244'
                      
                      return (
                        <div key={i}
                          className="flex items-center 
                                     justify-between px-2 py-1.5"
                          style={{ 
                            background: '#0a1520',
                            borderLeft: `2px solid ${color}`
                          }}>
                          
                          <div className="flex items-center gap-2">
                            <span className="font-exo text-[9px]"
                              style={{ color: '#3a6080' }}>
                              {new Date(op.fecha)
                                .toLocaleDateString('es-CL', {
                                  day: '2-digit',
                                  month: '2-digit'
                                })}

                            {op.origen && op.destino && (
                              <span className="font-exo text-[9px]"
                                style={{ color: '#c8e8ff' }}>
                                {op.origen.slice(0,20)} → {' '}
                                {op.destino.slice(0,20)}
                              </span>
                            )}
                          </div>
                          
                          <div className="flex items-center gap-3">
                            <span className="font-space text-[10px]"
                              style={{ color: '#c8e8ff' }}>
                              {op.km_ecu?.toFixed(0)}km
                            </span>
                            <span className="font-space text-[10px]"
                              style={{ color: '#ffcc00' }}>
                              {op.litros_ecu?.toFixed(0)}L
                            </span>
                            <span className="font-space text-[10px] 
                                             font-bold"
                              style={{ color: op.rendimiento >= 2.5
                                ? '#00ff88' : '#ffcc00' }}>
                              {op.rendimiento?.toFixed(2)} km/L
                            </span>
                            {op.balance !== null && (
                              <span className="font-space text-[9px]"
                                style={{ 
                                  color: Math.abs(op.balance) < 50
                                    ? '#00ff88'
                                    : Math.abs(op.balance) < 200
                                    ? '#ffcc00' : '#ff2244'
                                }}>
                                {op.balance > 0 ? '+' : ''}
                                {op.balance}L
                              </span>
                            )}
                            <span className="font-exo text-[7px] 
                                             px-1"
                              style={{ 
                                color,
                                border: `1px solid ${color}40`
                              }}>
                              {op.calidad}
                            </span>
                          </div>
                        </div>
                      )
                    })}
                  </div>
                </div>
              )}
              
              {data.operaciones.length === 0 && (
                <div className="py-8 text-center"
                  style={{ 
                    background: '#060d14',
                    border: '1px solid #0d2035'
                  }}>
                  <div className="font-exo text-[11px] mb-1"
                    style={{ color: '#3a6080' }}>
                    Sin operaciones verificadas este mes
                  </div>
                  <div className="font-exo text-[9px]"
                    style={{ color: '#1a3a55' }}>
                    Las operaciones aparecen cuando el sistema
                    cierra períodos entre cargas con datos ECU completos
                  </div>
                </div>
              )}
              
            </div>
          </>
        )}
      </div>
    </div>
  )
}
