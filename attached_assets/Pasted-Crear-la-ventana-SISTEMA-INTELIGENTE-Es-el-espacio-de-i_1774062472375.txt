async function getInconsistencias(req, res) {
  
  const items = []
  
  // 1. Alertas de patrón sin gestionar
  const alertasPatron = await db.select()
    .from(alertasAprendizaje)
    .where(eq(alertasAprendizaje.gestionado, false))
    .orderBy(desc(alertasAprendizaje.fecha))
    .limit(20)
  
  for (const a of alertasPatron) {
    items.push({
      id: `alerta_${a.id}`,
      tipo: 'ALERTA_PATRON',
      nivel: 'REVISAR',
      titulo: `${a.entidadNombre} — ${a.tipo}`,
      descripcion: a.descripcion,
      valor: a.diferenciaPct,
      contrato: a.contrato,
      fecha: a.fecha,
      acciones: ['CONFIRMAR', 'FALSA_ALARMA', 'AJUSTAR']
    })
  }
  
  // 2. Corredores sin geocodificar
  const corredoresSinNombre = await db.select()
    .from(corredores)
    .where(
      or(
        like(corredores.origenNombre, '-%'),
        eq(corredores.origenNombre, 'Punto desconocido')
      )
    )
    .limit(10)
  
  for (const c of corredoresSinNombre) {
    items.push({
      id: `corredor_${c.id}`,
      tipo: 'CORREDOR_SIN_NOMBRE',
      nivel: 'INFO',
      titulo: `Corredor sin geocodificar`,
      descripcion: `${c.origenNombre} → ${c.destinoNombre} · ${c.totalViajesBase} viajes · ${c.contrato}`,
      valor: c.totalViajesBase,
      contrato: c.contrato,
      fecha: c.updatedAt,
      acciones: ['ASIGNAR_NOMBRE', 'IGNORAR']
    })
  }
  
  // 3. Camiones con balance anómalo esta semana
  const hace7dias = new Date(Date.now() - 7*24*60*60*1000)
  const balancesAnomalos = await db.execute(sql`
    SELECT 
      patente,
      conductor,
      contrato,
      SUM(litros_cargados_total) as total_cargado,
      SUM(litros_consumidos_ecu) as total_consumido,
      COUNT(*) as periodos,
      ROUND(
        (SUM(litros_cargados_total) - SUM(litros_consumidos_ecu))
        / NULLIF(SUM(litros_consumidos_ecu), 0) * 100
      ) as balance_pct
    FROM entre_cargas
    WHERE carga_inicio_fecha >= ${hace7dias}
      AND litros_consumidos_ecu > 0
      AND litros_cargados_total > 0
    GROUP BY patente, conductor, contrato
    HAVING ABS(
      (SUM(litros_cargados_total) - SUM(litros_consumidos_ecu))
      / NULLIF(SUM(litros_consumidos_ecu), 0) * 100
    ) > 30
    ORDER BY ABS(balance_pct) DESC
    LIMIT 10
  `)
  
  for (const b of balancesAnomalos.rows) {
    items.push({
      id: `balance_${b.patente}`,
      tipo: 'BALANCE_ANOMALO',
      nivel: Math.abs(b.balance_pct) > 80 
        ? 'CRITICO' : 'SOSPECHOSO',
      titulo: `${b.patente} — Balance ${b.balance_pct > 0 ? '+' : ''}${b.balance_pct}%`,
      descripcion: `${b.conductor || 'Sin conductor'} · ${b.contrato} · Cargó ${Math.round(b.total_cargado)}L, consumió ${Math.round(b.total_consumido)}L en ${b.periodos} períodos esta semana`,
      valor: b.balance_pct,
      contrato: b.contrato,
      fecha: new Date(),
      acciones: ['INVESTIGAR', 'CONFIRMAR_NORMAL', 'ESCALAR']
    })
  }
  
  // 4. Parámetros con baja confianza
  // que llevan mucho tiempo sin mejorar
  const parametrosBajos = await db.select()
    .from(parametrosAdaptativos)
    .where(
      and(
        eq(parametrosAdaptativos.confianza, 'BAJA'),
        lt(parametrosAdaptativos.totalMuestras, 5)
      )
    )
    .limit(5)
  
  if (parametrosBajos.length > 0) {
    items.push({
      id: 'params_baja_confianza',
      tipo: 'PARAMETROS_INSUFICIENTES',
      nivel: 'INFO',
      titulo: `${parametrosBajos.length} parámetros con datos insuficientes`,
      descripcion: `El sistema necesita más viajes para calibrar estos corredores. Los umbrales actuales son estimados.`,
      valor: parametrosBajos.length,
      contrato: 'TODOS',
      fecha: new Date(),
      acciones: ['VER_DETALLE', 'AJUSTE_MANUAL']
    })
  }
  
  // Ordenar: CRITICO primero, luego SOSPECHOSO, INFO al fondo
  const orden = { CRITICO: 0, SOSPECHOSO: 1, REVISAR: 2, INFO: 3 }
  items.sort((a, b) => 
    (orden[a.nivel] || 3) - (orden[b.nivel] || 3)
  )
  
  res.json({
    total: items.length,
    criticos: items.filter(i => i.nivel === 'CRITICO').length,
    items
  })
}

app.get("/api/sistema/inconsistencias", getInconsistencias)

async function resolverInconsistencia(req, res) {
  const { id, tipo, accion, nota, valor_nuevo } = req.body
  
  if (tipo === 'ALERTA_PATRON') {
    const alertaId = parseInt(id.replace('alerta_', ''))
    
    await db.update(alertasAprendizaje)
      .set({ 
        gestionado: true,
        nota: nota || accion
      })
      .where(eq(alertasAprendizaje.id, alertaId))
    
    // Si es FALSA_ALARMA → recalibrar umbral
    if (accion === 'FALSA_ALARMA') {
      await aplicarFeedback({
        alertaTipo: 'CAMBIO_PATRON',
        decision: 'FALSA_ALARMA',
        nota
      })
    }
  }
  
  if (tipo === 'CORREDOR_SIN_NOMBRE' && 
      accion === 'ASIGNAR_NOMBRE') {
    const corredorId = parseInt(
      id.replace('corredor_', '')
    )
    await db.update(corredores)
      .set({ 
        origenNombre: valor_nuevo?.origen || 'Sin nombre',
        destinoNombre: valor_nuevo?.destino || 'Sin nombre'
      })
      .where(eq(corredores.id, corredorId))
  }
  
  res.json({ ok: true })
}

app.post("/api/sistema/resolver-inconsistencia",
  resolverInconsistencia)

function SistemaInteligente() {

    queryKey: ['/api/sistema/inconsistencias'],

  })

    queryKey: ['/api/aprendizaje/parametros']
  })

    queryKey: ['/api/sistema/reporte-semanal']
  })

              style={{ color: '#3a6080' }}>

function SeccionInconsistencias({ data, onResolver }) {
  
  const [resolviendo, setResolviendo] = 
    useState<string | null>(null)
  
  const resolver = async (
    item: any,
    accion: string,
    nota?: string,
    valor_nuevo?: any
  ) => {
    setResolviendo(item.id)
    
    await fetch('/api/sistema/resolver-inconsistencia', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        id: item.id,
        tipo: item.tipo,
        accion,
        nota,
        valor_nuevo
      })
    })
    
    setResolviendo(null)
    onResolver()
  }
  
  if (!data?.items?.length) {
    return (
      <div className="py-20 text-center">
        <div className="text-4xl mb-4">✅</div>
        <div className="font-space text-[14px] 
                        tracking-wider mb-2"
          style={{ color: '#00ff88' }}>
          SIN INCONSISTENCIAS ACTIVAS
        </div>
        <div className="font-exo text-[11px]"
          style={{ color: '#3a6080' }}>
          El sistema opera dentro de parámetros normales.
          Las nuevas inconsistencias aparecerán aquí
          automáticamente.
        </div>
      </div>
    )
  }
  
  const colores = {
    CRITICO:    '#ff2244',
    SOSPECHOSO: '#ff8c00',
    REVISAR:    '#ffcc00',
    INFO:       '#00d4ff'
  }
  
  return (
    <div className="space-y-3">
      
      {/* Resumen rápido */}
      <div className="grid grid-cols-4 gap-3 mb-4">
        {[
          { 
            label: 'CRÍTICOS', 
            n: data.items.filter(i => i.nivel === 'CRITICO').length,
            color: '#ff2244' 
          },
          { 
            label: 'SOSPECHOSOS', 
            n: data.items.filter(i => i.nivel === 'SOSPECHOSO').length,
            color: '#ff8c00' 
          },
          { 
            label: 'REVISAR', 
            n: data.items.filter(i => i.nivel === 'REVISAR').length,
            color: '#ffcc00' 
          },
          { 
            label: 'INFORMATIVOS', 
            n: data.items.filter(i => i.nivel === 'INFO').length,
            color: '#00d4ff' 
          },
        ].map(s => (
          <div key={s.label} className="p-3 text-center"
            style={{ 
              background: '#060d14',
              border: `1px solid ${s.color}20`,
              borderTop: `2px solid ${s.color}`
            }}>
            <div className="font-space text-[24px] 
                            font-bold"
              style={{ color: s.color }}>
              {s.n}
            </div>
            <div className="font-exo text-[8px]"
              style={{ color: '#3a6080' }}>
              {s.label}
            </div>
          </div>
        ))}
      </div>
      
      {/* Lista de inconsistencias */}
      {data.items.map(item => {
        const color = colores[item.nivel] || '#3a6080'
        
        return (
          <div key={item.id}
            className="border p-4"
            style={{ 
              borderColor: `${color}30`,
              background: `${color}05`,
              borderLeft: `3px solid ${color}`
            }}>
            
            {/* Header del item */}
            <div className="flex items-start 
                            justify-between gap-4">
              <div className="flex-1">
                
                <div className="flex items-center gap-2 mb-1">
                  <span className="font-exo text-[8px] 
                                   font-bold px-2 py-0.5"
                    style={{ 
                      color,
                      border: `1px solid ${color}40`,
                      background: `${color}15`
                    }}>
                    {item.nivel}
                  </span>
                  <span className="font-exo text-[8px]"
                    style={{ color: '#3a6080' }}>
                    {item.tipo.replace(/_/g, ' ')}
                  </span>
                  {item.contrato && (
                    <span className="font-exo text-[8px] 
                                     px-1.5"
                      style={{ 
                        color: '#00d4ff',
                        border: '1px solid #00d4ff20'
                      }}>
                      {item.contrato}
                    </span>
                  )}
                </div>
                
                <div className="font-space text-[12px] 
                                font-bold mb-1"
                  style={{ color: '#c8e8ff' }}>
                  {item.titulo}
                </div>
                
                <div className="font-rajdhani text-[12px]"
                  style={{ color: '#3a6080' }}>
                  {item.descripcion}
                </div>
              </div>
              
              {/* Valor destacado */}
              {item.valor !== null && (
                <div className="text-right flex-shrink-0">
                  <div className="font-space text-[22px] 
                                  font-bold"
                    style={{ color }}>
                    {typeof item.valor === 'number' 
                      ? item.valor > 0 ? `+${item.valor}` 
                        : item.valor
                      : item.valor}
                    {item.tipo === 'BALANCE_ANOMALO' 
                      ? '%' : ''}
                  </div>
                </div>
              )}
            </div>
            
            {/* Botones de acción */}
            <div className="flex items-center gap-2 mt-3">
              {item.acciones.map(accion => (
                <button
                  key={accion}
                  onClick={() => {
                    if (accion === 'FALSA_ALARMA' ||
                        accion === 'CONFIRMAR_NORMAL') {
                      const nota = window.prompt(
                        'Motivo (opcional):'
                      )
                      resolver(item, accion, nota || undefined)
                    } else if (accion === 'AJUSTAR' ||
                               accion === 'ASIGNAR_NOMBRE') {
                      const valor = window.prompt(
                        'Nuevo valor:'
                      )
                      if (valor) resolver(
                        item, accion, undefined, 
                        { valor }
                      )
                    } else {
                      resolver(item, accion)
                    }
                  }}
                  disabled={resolviendo === item.id}
                  className="font-exo text-[9px] font-bold
                             px-3 py-1.5 border cursor-pointer
                             transition-all hover:opacity-80
                             disabled:opacity-40"
                  style={{
                    borderColor: accion === 'CONFIRMAR' ||
                                 accion === 'CONFIRMAR_NORMAL'
                      ? '#00ff8840'
                      : accion === 'FALSA_ALARMA' ||
                        accion === 'IGNORAR'
                      ? '#3a608040'
                      : accion === 'ESCALAR' ||
                        accion === 'INVESTIGAR'
                      ? '#ff224440'
                      : '#ff6b3540',
                    color: accion === 'CONFIRMAR' ||
                           accion === 'CONFIRMAR_NORMAL'
                      ? '#00ff88'
                      : accion === 'FALSA_ALARMA' ||
                        accion === 'IGNORAR'
                      ? '#3a6080'
                      : accion === 'ESCALAR' ||
                        accion === 'INVESTIGAR'
                      ? '#ff2244'
                      : '#ff6b35'
                  }}>
                  {resolviendo === item.id 
                    ? '...' 
                    : accion.replace(/_/g, ' ')}
                </button>
              ))}
            </div>
          </div>
        )
      })}
    </div>
  )
}

function SeccionCalibracion({ parametros }) {
  
  const [editando, setEditando] = 
    useState<string | null>(null)
  const [nuevoValor, setNuevoValor] = useState('')
  
  const aplicarAjuste = async (
    scopeTipo: string,
    scopeId: string,
    parametro: string
  ) => {
    await fetch('/api/feedback/alerta', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        alertaTipo: 'AJUSTE_MANUAL_CEO',
        entidadTipo: scopeTipo,
        entidadId: scopeId,
        contrato: '',
        decision: 'AJUSTAR',
        parametroAfectado: parametro,
        ajusteManual: parseFloat(nuevoValor)
      })
    })
    setEditando(null)
    setNuevoValor('')
  }
  
  if (!parametros) return (
    <div className="py-10 text-center font-exo text-[11px]"
      style={{ color: '#3a6080' }}>
      Cargando parámetros...
    </div>
  )
  
  return (
    <div className="space-y-4">
      
      {/* Resumen de calibración */}
      <div className="p-4 border"
        style={{ 
          borderColor: '#0d2035',
          background: '#060d14'
        }}>
        <div className="font-exo text-[8px] tracking-wider
                        uppercase mb-3"
          style={{ color: '#3a6080' }}>
          ESTADO DE CALIBRACIÓN GLOBAL

function SeccionDialogoIA({ inconsistencias }) {
  
  const [mensajes, setMensajes] = useState<{
    rol: 'sistema' | 'ceo'
    texto: string
    timestamp: Date
  }[]>([])
  const [input, setInput] = useState('')
  const [cargando, setCargando] = useState(false)
  
  // Mensaje inicial automático al abrir
  useEffect(() => {
    if (mensajes.length === 0) {
      cargarMensajeInicial()
    }
  }, [])
  
  const cargarMensajeInicial = async () => {
    setCargando(true)
    
    const res = await fetch('/api/sistema/resumen-ia')
    const data = await res.json()
    
    setMensajes([{
      rol: 'sistema',
      texto: data.resumen || 
        'Sistema inicializando...',
      timestamp: new Date()
    }])
    
    setCargando(false)
  }
  
  const enviar = async () => {
    if (!input.trim()) return
    
    const pregunta = input
    setInput('')
    
    setMensajes(prev => [...prev, {
      rol: 'ceo',
      texto: pregunta,
      timestamp: new Date()
    }])
    
    setCargando(true)
    
    try {
      const res = await fetch('/api/sistema/consulta', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ 
          pregunta,
          contexto: 'sistema_inteligente'
        })
      })
      const data = await res.json()
      
      setMensajes(prev => [...prev, {
        rol: 'sistema',
        texto: data.respuesta,
        timestamp: new Date()
      }])
    } catch (e) {
      setMensajes(prev => [...prev, {
        rol: 'sistema',
        texto: 'Error al conectar con el sistema.',
        timestamp: new Date()
      }])
    }
    
    setCargando(false)
  }
  
  // Sugerencias basadas en inconsistencias activas
  const sugerencias = [
    inconsistencias?.items?.find(
      i => i.tipo === 'BALANCE_ANOMALO'
    )
      ? `¿Por qué ${inconsistencias.items.find(
          i => i.tipo === 'BALANCE_ANOMALO'
        )?.titulo}?`
      : null,
    '¿Qué aprendiste esta semana?',
    '¿En qué corredor tienes más confianza?',
    '¿Qué necesitas que yo revise?',
    '¿Cuáles alertas fueron falsas alarmas?',
  ].filter(Boolean)
  
  return (
    <div className="space-y-4">
      
      {/* Historial de mensajes */}
      <div className="border p-4 space-y-3 
                      min-h-[300px] max-h-[400px] 
                      overflow-y-auto"
        style={{ 
          borderColor: '#0d2035',
          background: '#060d14'
        }}>
        
        {cargando && mensajes.length === 0 && (
          <div className="flex items-center gap-2 py-4">
            <Loader2 className="w-4 h-4 animate-spin"
              style={{ color: '#ff6b35' }} />
            <span className="font-exo text-[10px]"
              style={{ color: '#3a6080' }}>
              El sistema está preparando su resumen...
            </span>
          </div>
        )}
        
        {mensajes.map((msg, i) => (
          <div key={i}
            className={`flex ${msg.rol === 'ceo' 
              ? 'justify-end' : 'justify-start'}`}>
            <div className="max-w-[80%] p-3"
              style={{
                background: msg.rol === 'ceo'
                  ? 'rgba(255,107,53,0.08)'
                  : '#0a1520',
                borderLeft: msg.rol === 'sistema'
                  ? '2px solid #ff6b35' : 'none',
                borderRight: msg.rol === 'ceo'
                  ? '2px solid #ff6b35' : 'none'
              }}>
              <div className="font-exo text-[8px] mb-1"
                style={{ color: '#3a6080' }}>
                {msg.rol === 'sistema' 
                  ? '🧠 SISTEMA ADAPTATIVO'
                  : '👤 CEO'}
                {' · '}
                {msg.timestamp.toLocaleTimeString(
                  'es-CL', { 
                    hour: '2-digit', 
                    minute: '2-digit' 
                  }
                )}
              </div>
              <div className="font-rajdhani text-[12px]
                              leading-relaxed"
                style={{ color: '#c8e8ff' }}>
                {msg.texto}
              </div>
            </div>
          </div>
        ))}
        
        {cargando && mensajes.length > 0 && (
          <div className="flex items-center gap-2">
            <Loader2 className="w-3 h-3 animate-spin"
              style={{ color: '#ff6b35' }} />
            <span className="font-exo text-[9px]"
              style={{ color: '#3a6080' }}>
              Analizando...
            </span>
          </div>
        )}
      </div>
      
      {/* Sugerencias */}
      <div className="flex flex-wrap gap-2">
        {sugerencias.map((s, i) => (
          <button key={i}
            onClick={() => {
              setInput(s)
            }}
            className="font-exo text-[9px] px-3 py-1.5
                       border cursor-pointer transition-all
                       hover:border-[#ff6b35]"
            style={{ 
              borderColor: '#0d2035',
              color: '#3a6080'
            }}>
            {s}
          </button>
        ))}
      </div>
      
      {/* Input */}
      <div className="flex gap-2">
        <input
          value={input}
          onChange={e => setInput(e.target.value)}
          onKeyDown={e => e.key === 'Enter' && enviar()}
          placeholder="Pregunta al sistema o pídele que explique algo..."
          className="flex-1 px-4 py-3 bg-transparent 
                     border outline-none font-rajdhani 
                     text-[13px] transition-all"
          style={{ 
            borderColor: '#0d2035',
            color: '#c8e8ff'
          }}
          disabled={cargando}
        />
        <button
          onClick={enviar}
          disabled={cargando || !input.trim()}
          className="px-5 py-3 font-space text-[11px]
                     font-bold cursor-pointer border
                     transition-all hover:opacity-80
                     disabled:opacity-40"
          style={{ 
            borderColor: '#ff6b35',
            color: '#ff6b35',
            background: 'rgba(255,107,53,0.08)'
          }}>
          ENVIAR
        </button>
      </div>
    </div>
  )
}

function SeccionReporte({ reporte }) {
  
  const [generando, setGenerando] = useState(false)
  
  const generarReporte = async () => {
    setGenerando(true)
    await fetch('/api/sistema/generar-reporte-semanal', {
      method: 'POST'
    })
    setGenerando(false)
    // El query se refresca automáticamente
  }
  
  return (
    <div className="space-y-4">
      
      <div className="flex items-center justify-between">
        <div>
          <div className="font-exo text-[8px] 
                          tracking-wider uppercase"
            style={{ color: '#3a6080' }}>
            REPORTE SEMANAL DEL SISTEMA
          </div>
          {reporte?.fecha && (
            <div className="font-exo text-[10px] mt-1"
              style={{ color: '#3a6080' }}>
              Generado: {new Date(reporte.fecha)
                .toLocaleDateString('es-CL', {
                  weekday: 'long',
                  day: 'numeric',
                  month: 'long'
                })}
            </div>
          )}
        </div>
        <button
          onClick={generarReporte}
          disabled={generando}
          className="font-exo text-[10px] px-4 py-2
                     border cursor-pointer transition-all
                     hover:border-[#ff6b35]
                     disabled:opacity-40"
          style={{ 
            borderColor: '#0d2035',
            color: '#3a6080'
          }}>
          {generando ? (
            <span className="flex items-center gap-2">
              <Loader2 className="w-3 h-3 animate-spin" />
              Generando...
            </span>
          ) : '🔄 Generar nuevo'}
        </button>
      </div>
      
      {!reporte ? (
        <div className="py-12 text-center border"
          style={{ 
            borderColor: '#0d2035',
            background: '#060d14'
          }}>
          <div className="font-exo text-[11px] mb-2"
            style={{ color: '#3a6080' }}>
            Sin reporte generado esta semana
          </div>
          <button
            onClick={generarReporte}
            className="font-exo text-[10px] px-4 py-2
                       border cursor-pointer mt-2"
            style={{ 
              borderColor: '#ff6b3540',
              color: '#ff6b35'
            }}>
            Generar ahora
          </button>
        </div>
      ) : (
        <div className="space-y-3">
          {[
            { key: 'aprendi', icon: '📚', 
              titulo: 'QUÉ APRENDÍ ESTA SEMANA' },
            { key: 'corregi', icon: '✅', 
              titulo: 'QUÉ CORREGÍ' },
            { key: 'preocupa', icon: '⚠️', 
              titulo: 'QUÉ ME PREOCUPA' },
            { key: 'necesito', icon: '🤝', 
              titulo: 'QUÉ NECESITO QUE REVISES' },
            { key: 'proximos', icon: '🎯', 
              titulo: 'PRÓXIMOS PASOS' },
          ].map(seccion => (
            reporte[seccion.key] && (
              <div key={seccion.key}
                className="p-4 border"
                style={{ 
                  borderColor: '#0d2035',
                  background: '#060d14',
                  borderLeft: '3px solid #ff6b35'
                }}>
                <div className="font-exo text-[8px] 
                                tracking-wider uppercase mb-2"
                  style={{ color: '#ff6b35' }}>
                  {seccion.icon} {seccion.titulo}
                </div>
                <div className="font-rajdhani text-[13px]
                                leading-relaxed"
                  style={{ color: '#c8e8ff' }}>
                  {reporte[seccion.key]}
                </div>
              </div>
            )
          ))}
        </div>
      )}
    </div>
  )
}
