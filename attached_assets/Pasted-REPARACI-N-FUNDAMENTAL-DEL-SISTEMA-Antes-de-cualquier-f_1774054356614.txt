import { db } from "../db"
import { camiones } from "../../shared/schema"
import { isNotNull } from "drizzle-orm"

// Cache en memoria — recalcular cada hora
let _vinAPatente: Map<string, string> | null = null
let _patenteAVin: Map<string, string> | null = null
let _cacheTime = 0
const CACHE_TTL = 60 * 60 * 1000 // 1 hora

export async function getVinPatente(): 
  Promise<Map<string, string>> {
  await refreshCacheIfNeeded()
  return _vinAPatente!
}

export async function getPatenteVin(): 
  Promise<Map<string, string>> {
  await refreshCacheIfNeeded()
  return _patenteAVin!
}

async function refreshCacheIfNeeded() {
  if (_vinAPatente && Date.now() - _cacheTime < CACHE_TTL) 
    return
  
  const todos = await db.select()
    .from(camiones)
    .where(isNotNull(camiones.vin))
  
  // Agrupar por VIN
  const porVin: Record<string, typeof todos> = {}
  for (const c of todos) {
    if (!c.vin) continue
    if (!porVin[c.vin]) porVin[c.vin] = []
    porVin[c.vin].push(c)
  }
  
  const vinAPatente = new Map<string, string>()
  const patenteAVin = new Map<string, string>()
  
  for (const [vin, registros] of Object.entries(porVin)) {
    
    // Regla de selección — en orden de prioridad:
    // 1. Patente alfanumérica nueva (XXXX00)
    // 2. Patente con letras (cualquier formato)
    // 3. La que sea si no hay otra opción
    
    const alfanumerica = registros.find(r => 
      /^[A-Z]{4}[0-9]{2}$/.test(r.patente || '')
    )
    const conLetras = registros.find(r =>
      /[A-Z]/.test(r.patente || '')
    )
    const cualquiera = registros[0]
    
    const elegida = alfanumerica || conLetras || cualquiera
    
    if (elegida?.patente) {
      vinAPatente.set(vin, elegida.patente)
      patenteAVin.set(elegida.patente, vin)
      
      // También mapear patentes alternativas → mismo VIN
      for (const r of registros) {
        if (r.patente && r.patente !== elegida.patente) {
          patenteAVin.set(r.patente, vin)
        }
      }
    }
  }
  
  _vinAPatente = vinAPatente
  _patenteAVin = patenteAVin
  _cacheTime = Date.now()
  
  console.log(
    `[VIN-PATENTE] Cache actualizado: ` +
    `${vinAPatente.size} VINs resueltos, ` +
    `${patenteAVin.size} patentes mapeadas`
  )
}

// Helper: dado un VIN obtener patente activa
export async function vinAPatente(vin: string): 
  Promise<string | null> {
  const map = await getVinPatente()
  return map.get(vin) || null
}

// Helper: dada una patente obtener VIN
export async function patenteAVin(patente: string): 
  Promise<string | null> {
  const map = await getPatenteVin()
  return map.get(patente) || null
}

// Diagnóstico: mostrar el estado del mapeo
export async function diagnosticoVinPatente() {
  const map = await getVinPatente()
  const todos = await db.select()
    .from(camiones)
    .where(isNotNull(camiones.vin))
  
  const total = new Set(todos.map(c => c.vin)).size
  const resueltos = map.size
  const sinResolver = total - resueltos
  
  return {
    total_vins: total,
    vins_resueltos: resueltos,
    vins_sin_resolver: sinResolver,
    cobertura_pct: Math.round(resueltos / total * 100),
    muestra: Array.from(map.entries()).slice(0, 5)
      .map(([vin, pat]) => ({ vin, patente: pat }))
  }
}

import { db } from "../db"
import { volvoFuelSnapshots } from "../../shared/schema"
import { and, eq, gte, lte, desc, asc } from "drizzle-orm"
import { patenteAVin } from "./vin-patente"

export interface SnapshotsCarga {
  vin: string | null
  
  // Snapshot más cercano ANTES de la carga
  // (representa el estado del tanque antes de cargar)
  snapAntes: {
    capturedAt: Date
    totalFuelUsedMl: number    // mL acumulados
    totalDistanceM: number     // metros acumulados
    fuelLevelPct: number | null // % tanque si disponible
    minutosAntes: number        // cuánto antes de la carga
  } | null
  
  // Snapshot más cercano DESPUÉS de la carga
  // (el camión ya cargó y reanudó el viaje)
  snapDespues: {
    capturedAt: Date
    totalFuelUsedMl: number
    totalDistanceM: number
    fuelLevelPct: number | null
    minutosDespues: number
  } | null
  
  // Calidad del cruce
  calidadCruce: 'EXCELENTE' | 'BUENA' | 'REGULAR' | 'MALA' | 'SIN_DATOS'
  // EXCELENTE: snap antes y después dentro de 15 min
  // BUENA: ambos dentro de 30 min
  // REGULAR: ambos dentro de 60 min
  // MALA: solo uno de los dos
  // SIN_DATOS: ninguno encontrado
}

export async function encontrarSnapshotsParaCarga(
  patente: string,
  fechaCarga: Date,
  ventanaMinutos: number = 60
): Promise<SnapshotsCarga> {
  
  // 1. Resolver VIN
  const vin = await patenteAVin(patente)
  
  if (!vin) {
    return {
      vin: null,
      snapAntes: null,
      snapDespues: null,
      calidadCruce: 'SIN_DATOS'
    }
  }
  
  const ventana = ventanaMinutos * 60 * 1000
  const desde = new Date(fechaCarga.getTime() - ventana)
  const hasta = new Date(fechaCarga.getTime() + ventana)
  
  // 2. Obtener snapshots en la ventana
  const snapshots = await db.select()
    .from(volvoFuelSnapshots)
    .where(
      and(
        eq(volvoFuelSnapshots.vin, vin),
        gte(volvoFuelSnapshots.capturedAt, desde),
        lte(volvoFuelSnapshots.capturedAt, hasta)
      )
    )
    .orderBy(asc(volvoFuelSnapshots.capturedAt))
  
  if (snapshots.length === 0) {
    return { vin, snapAntes: null, snapDespues: null, 
             calidadCruce: 'SIN_DATOS' }
  }
  
  // 3. Encontrar el más cercano ANTES
  const antesDeCargar = snapshots
    .filter(s => new Date(s.capturedAt) <= fechaCarga)
    .sort((a, b) => 
      new Date(b.capturedAt).getTime() - 
      new Date(a.capturedAt).getTime()
    )
  
  // 4. Encontrar el más cercano DESPUÉS
  // Esperar al menos 20 min después de la carga
  // para que el camión haya reanudado el viaje
  const minEsperaDespues = new Date(
    fechaCarga.getTime() + 20 * 60 * 1000
  )
  const despuesDeCargar = snapshots
    .filter(s => new Date(s.capturedAt) >= minEsperaDespues)
    .sort((a, b) => 
      new Date(a.capturedAt).getTime() - 
      new Date(b.capturedAt).getTime()
    )
  
  const mejor_antes = antesDeCargar[0] || null
  const mejor_despues = despuesDeCargar[0] || null
  
  // 5. Calcular calidad del cruce
  let calidadCruce: SnapshotsCarga['calidadCruce'] = 'SIN_DATOS'
  
  if (mejor_antes && mejor_despues) {
    const minAntes = (fechaCarga.getTime() - 
      new Date(mejor_antes.capturedAt).getTime()) / 60000
    const minDespues = (new Date(mejor_despues.capturedAt)
      .getTime() - fechaCarga.getTime()) / 60000
    const maxMin = Math.max(minAntes, minDespues)
    
    if (maxMin <= 15) calidadCruce = 'EXCELENTE'
    else if (maxMin <= 30) calidadCruce = 'BUENA'
    else if (maxMin <= 60) calidadCruce = 'REGULAR'
    else calidadCruce = 'MALA'
  } else if (mejor_antes || mejor_despues) {
    calidadCruce = 'MALA'
  }
  
  return {
    vin,
    snapAntes: mejor_antes ? {
      capturedAt: new Date(mejor_antes.capturedAt),
      totalFuelUsedMl: mejor_antes.totalFuelUsed || 0,
      totalDistanceM: mejor_antes.totalDistance || 0,
      fuelLevelPct: mejor_antes.fuelLevel || null,
      minutosAntes: Math.round(
        (fechaCarga.getTime() - 
         new Date(mejor_antes.capturedAt).getTime()) / 60000
      )
    } : null,
    snapDespues: mejor_despues ? {
      capturedAt: new Date(mejor_despues.capturedAt),
      totalFuelUsedMl: mejor_despues.totalFuelUsed || 0,
      totalDistanceM: mejor_despues.totalDistance || 0,
      fuelLevelPct: mejor_despues.fuelLevel || null,
      minutosDespues: Math.round(
        (new Date(mejor_despues.capturedAt).getTime() - 
         fechaCarga.getTime()) / 60000
      )
    } : null,
    calidadCruce
  }
}

// Para el análisis ENTRE CARGAS:
// Dado dos cargas consecutivas, calcular
// todo lo que pasó entre ellas

export interface PeriodoEntreCargas {
  vin: string
  
  // Métricas del período
  kmEcu: number | null           // km reales ECU
  litrosConsumidosEcu: number | null  // L quemados ECU
  rendimientoEcu: number | null  // km/L real
  horasPeriodo: number | null
  
  // Estado del tanque
  nivelTanqueInicioPct: number | null  // % antes de carga A
  nivelTanqueFinPct: number | null     // % antes de carga B
  
  // Calidad
  calidadDatos: 'COMPLETO' | 'PARCIAL' | 'SIN_ECU'
  snapCount: number  // cuántos snapshots hay entre A y B
}

export async function calcularPeriodoEntreCargas(
  patente: string,
  cargaA: { fecha: Date, litros: number },
  cargaB: { fecha: Date, litros: number } | null
): Promise<PeriodoEntreCargas> {
  
  const vin = await patenteAVin(patente)
  
  if (!vin) {
    return {
      vin: '',
      kmEcu: null,
      litrosConsumidosEcu: null,
      rendimientoEcu: null,
      horasPeriodo: cargaB 
        ? (cargaB.fecha.getTime() - cargaA.fecha.getTime()) 
          / 3600000
        : null,
      nivelTanqueInicioPct: null,
      nivelTanqueFinPct: null,
      calidadDatos: 'SIN_ECU',
      snapCount: 0
    }
  }
  
  // Ventana: desde 30min antes de carga A
  // hasta 30min después de carga B (o +2h si no hay B)
  const inicioVentana = new Date(
    cargaA.fecha.getTime() - 30 * 60 * 1000
  )
  const finVentana = cargaB
    ? new Date(cargaB.fecha.getTime() + 30 * 60 * 1000)
    : new Date(cargaA.fecha.getTime() + 26 * 60 * 60 * 1000)
  
  const snapshotsPeriodo = await db.select()
    .from(volvoFuelSnapshots)
    .where(
      and(
        eq(volvoFuelSnapshots.vin, vin),
        gte(volvoFuelSnapshots.capturedAt, inicioVentana),
        lte(volvoFuelSnapshots.capturedAt, finVentana)
      )
    )
    .orderBy(asc(volvoFuelSnapshots.capturedAt))
  
  if (snapshotsPeriodo.length < 2) {
    return {
      vin,
      kmEcu: null,
      litrosConsumidosEcu: null,
      rendimientoEcu: null,
      horasPeriodo: cargaB
        ? (cargaB.fecha.getTime() - cargaA.fecha.getTime())
          / 3600000
        : null,
      nivelTanqueInicioPct: null,
      nivelTanqueFinPct: null,
      calidadDatos: 'SIN_ECU',
      snapCount: 0
    }
  }
  
  // Snapshot de referencia de inicio:
  // El primero DESPUÉS de carga A (ya cargó)
  const snapInicio = snapshotsPeriodo
    .filter(s => new Date(s.capturedAt) >= 
      new Date(cargaA.fecha.getTime() + 15 * 60 * 1000)
    )[0] || snapshotsPeriodo[0]
  
  // Snapshot de referencia de fin:
  // El más cercano ANTES de carga B
  const snapFin = cargaB
    ? snapshotsPeriodo
        .filter(s => new Date(s.capturedAt) <= cargaB.fecha)
        .sort((a, b) => 
          new Date(b.capturedAt).getTime() - 
          new Date(a.capturedAt).getTime()
        )[0]
    : snapshotsPeriodo[snapshotsPeriodo.length - 1]
  
  if (!snapInicio || !snapFin || snapInicio === snapFin) {
    return {
      vin,
      kmEcu: null,
      litrosConsumidosEcu: null,
      rendimientoEcu: null,
      horasPeriodo: null,
      nivelTanqueInicioPct: snapInicio?.fuelLevel || null,
      nivelTanqueFinPct: null,
      calidadDatos: 'PARCIAL',
      snapCount: snapshotsPeriodo.length
    }
  }
  
  // Calcular métricas
  // totalFuelUsed es ACUMULATIVO en mL
  // La diferencia = litros quemados en el período
  const fuelDeltaMl = 
    (snapFin.totalFuelUsed || 0) - 
    (snapInicio.totalFuelUsed || 0)
  
  const distDeltaM = 
    (snapFin.totalDistance || 0) - 
    (snapInicio.totalDistance || 0)
  
  const litrosConsumidos = Math.max(0, fuelDeltaMl / 1000)
  const kmRecorridos = Math.max(0, distDeltaM / 1000)
  
  const rendimiento = litrosConsumidos > 5 && kmRecorridos > 10
    ? Math.round((kmRecorridos / litrosConsumidos) * 100) / 100
    : null
  
  const horasPeriodo = 
    (new Date(snapFin.capturedAt).getTime() - 
     new Date(snapInicio.capturedAt).getTime()) / 3600000
  
  // Calidad de datos
  const calidadDatos = 
    litrosConsumidos > 0 && kmRecorridos > 0 
      ? 'COMPLETO' : 'PARCIAL'
  
  return {
    vin,
    kmEcu: kmRecorridos > 0 
      ? Math.round(kmRecorridos * 10) / 10 : null,
    litrosConsumidosEcu: litrosConsumidos > 0
      ? Math.round(litrosConsumidos * 10) / 10 : null,
    rendimientoEcu: rendimiento,
    horasPeriodo: Math.round(horasPeriodo * 10) / 10,
    nivelTanqueInicioPct: snapInicio.fuelLevel || null,
    nivelTanqueFinPct: snapFin.fuelLevel || null,
    calidadDatos,
    snapCount: snapshotsPeriodo.length
  }
}

app.post("/api/admin/recalcular-cuadratura",
async (req, res) => {
  const { calcularPeriodoEntreCargas } = 
    await import("../utils/snapshots-carga")
  const { getPatenteVin } = 
    await import("../utils/vin-patente")
  
  const patenteVin = await getPatenteVin()
  
  // Obtener viajes sin cruzar
  const viajesSinCruzar = await db.select()
    .from(viajesAprendizaje)
    .where(eq(viajesAprendizaje.sigetraCruzado, false))
    .limit(100) // de a 100 por llamada
  
  let actualizados = 0
  let sinVin = 0
  let sinCargas = 0
  
  for (const viaje of viajesSinCruzar) {
    const patente = viaje.patente
    if (!patente) continue
    
    const vin = patenteVin.get(patente)
    if (!vin) { sinVin++; continue }
    
    // Buscar cargas Sigetra cercanas al viaje
    const cargasViaje = await db.select()
      .from(cargas)
      .where(
        and(
          eq(cargas.patente, patente),
          gte(cargas.fecha, 
            new Date(
              new Date(viaje.fechaInicio!).getTime() 
              - 2 * 60 * 60 * 1000
            )
          ),
          lte(cargas.fecha,
            new Date(
              new Date(viaje.fechaFin || viaje.fechaInicio!)
                .getTime() + 2 * 60 * 60 * 1000
            )
          )
        )
      )
      .orderBy(asc(cargas.fecha))
    
    if (cargasViaje.length === 0) { sinCargas++; continue }
    
    const totalLitrosSigetra = cargasViaje
      .reduce((s, c) => s + (c.litros || 0), 0)
    
    if (totalLitrosSigetra === 0) continue
    
    const litrosEcu = viaje.litrosConsumidosEcu || 0
    
    if (litrosEcu > 0) {
      const delta = (
        (litrosEcu - totalLitrosSigetra) / 
        totalLitrosSigetra * 100
      )
      
      await db.update(viajesAprendizaje)
        .set({
          sigetraCruzado: true,
          deltaQuadratura: Math.round(delta * 10) / 10,
          litrosSigetraViajes: totalLitrosSigetra
        })
        .where(eq(viajesAprendizaje.id, viaje.id))
      
      actualizados++
    }
  }
  
  res.json({
    procesados: viajesSinCruzar.length,
    actualizados,
    sin_vin: sinVin,
    sin_cargas: sinCargas,
    pendientes_totales: await db.select({ count: count() })
      .from(viajesAprendizaje)
      .where(eq(viajesAprendizaje.sigetraCruzado, false))
      .then(r => r[0].count)
  })
})
