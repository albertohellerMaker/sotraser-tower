GET /api/ceo/kpis?periodo=7d
  camiones_total: COUNT camiones activos
  camiones_activos: COUNT volvo_ecu últimas 6h
  rendimiento_promedio: AVG km/L volvo_ecu período
  rendimiento_cambio_pct: vs período anterior igual duración
  litros_periodo: SUM cargas_tarjeta período
  litros_cambio_pct: vs período anterior
  alertas_criticas: COUNT alertas nivel CRITICO activas

GET /api/ceo/contratos-estado
  Para cada faena con camiones asignados:
  nombre, camiones_total, camiones_activos,
  rendimiento_promedio, alertas_criticas,
  estado: NORMAL | ATENCION | CRITICO
  
  CRITICO = alertas_criticas >= 3
  ATENCION = alertas_criticas 1-2 O rend < P50
  NORMAL = resto

  Array de {fecha, litros_dia, rendimiento_dia}
