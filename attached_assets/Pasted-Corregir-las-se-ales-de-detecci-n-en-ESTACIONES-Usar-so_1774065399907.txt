async function getUmbralCarga(
  patente: string,
  estacion: string,
  contrato: string,
  tipoUmbral: 'critico' | 'sospechoso' | 'revisar'
): Promise<number> {

  // PRIORIDAD 1: Patrón específico camión+estación
  const patronEspecifico = await db.select()
    .from(patronesCargaCombustible)
    .where(
      and(
        eq(patronesCargaCombustible.scopeTipo, 
           'CAMION_ESTACION'),
        eq(patronesCargaCombustible.patente, patente),
        eq(patronesCargaCombustible.estacionNombre, 
           estacion)
      )
    )
    .limit(1)

  if (patronEspecifico.length > 0 &&
      patronEspecifico[0].totalCargas >= 5) {
    const p = patronEspecifico[0]
    const prom = p.litrosPromedio
    if (tipoUmbral === 'critico')    return prom * 0.35
    if (tipoUmbral === 'sospechoso') return prom * 0.55
    if (tipoUmbral === 'revisar')    return prom * 0.70
  }

  // PRIORIDAD 2: Patrón del camión (todas estaciones)
  const patronCamion = await db.select()
    .from(patronesCargaCombustible)
    .where(
      and(
        eq(patronesCargaCombustible.scopeTipo, 'CAMION'),
        eq(patronesCargaCombustible.patente, patente)
      )
    )
    .limit(1)

  if (patronCamion.length > 0 &&
      patronCamion[0].totalCargas >= 3) {
    const p = patronCamion[0]
    const prom = p.litrosPromedio
    if (tipoUmbral === 'critico')    return prom * 0.35
    if (tipoUmbral === 'sospechoso') return prom * 0.55
    if (tipoUmbral === 'revisar')    return prom * 0.70
  }

  // PRIORIDAD 3: Parámetro adaptativo del contrato
  const paramContrato = await db.select()
    .from(parametrosAdaptativos)
    .where(
      and(
        eq(parametrosAdaptativos.scopeTipo, 'CONTRATO'),
        eq(parametrosAdaptativos.scopeId, contrato),
        eq(parametrosAdaptativos.parametro, 
           'litros_carga_minimo')
      )
    )
    .limit(1)

  if (paramContrato.length > 0 &&
      paramContrato[0].totalMuestras >= 10) {
    const umbral = paramContrato[0].umbralAnomalia
    if (tipoUmbral === 'critico')    return umbral * 0.5
    if (tipoUmbral === 'sospechoso') return umbral * 0.7
    if (tipoUmbral === 'revisar')    return umbral * 0.85
  }

  // PRIORIDAD 4: Parámetro global de la flota
  const paramGlobal = await db.select()
    .from(parametrosAdaptativos)
    .where(
      and(
        eq(parametrosAdaptativos.scopeTipo, 'GLOBAL'),
        eq(parametrosAdaptativos.scopeId, 'global'),
        eq(parametrosAdaptativos.parametro,
           'litros_carga_minimo')
      )
    )
    .limit(1)

  if (paramGlobal.length > 0) {
    const umbral = paramGlobal[0].umbralAnomalia
    if (tipoUmbral === 'critico')    return umbral * 0.5
    if (tipoUmbral === 'sospechoso') return umbral * 0.7
    if (tipoUmbral === 'revisar')    return umbral * 0.85
  }

  // PRIORIDAD 5: Sin datos suficientes
  // Retornar null → no generar señal todavía
  return null
}

async function getUmbralEntreCargas(
  contrato: string
): Promise<number> {

  const param = await db.select()
    .from(parametrosAdaptativos)
    .where(
      and(
        eq(parametrosAdaptativos.scopeTipo, 'CONTRATO'),
        eq(parametrosAdaptativos.scopeId, contrato),
        eq(parametrosAdaptativos.parametro,
           'horas_minimas_entre_cargas')
      )
    )
    .limit(1)

  if (param.length > 0 && 
      param[0].totalMuestras >= 10) {
    // Usar el percentil 10 como mínimo esperado
    // Si las cargas normalmente están separadas
    // por 14h, el p10 podría ser 6h
    // Dos cargas en menos de ese tiempo = raro
    return param[0].umbralAnomalia || 6
  }

  // Sin datos → usar parámetro global
  const global = await db.select()
    .from(parametrosAdaptativos)
    .where(
      and(
        eq(parametrosAdaptativos.scopeTipo, 'GLOBAL'),
        eq(parametrosAdaptativos.parametro,
           'horas_minimas_entre_cargas')
      )
    )
    .limit(1)

  if (global.length > 0) {
    return global[0].umbralAnomalia || 6
  }

  // Sin ningún parámetro aprendido
  // → no evaluar esta señal todavía
  return null
}

async function getUmbralEstanqueLleno(
  patente: string,
  contrato: string
): Promise<number | null> {

  // Buscar nivel típico de estanque antes
  // de cargar para este camión específico
  const param = await db.select()
    .from(parametrosAdaptativos)
    .where(
      and(
        eq(parametrosAdaptativos.scopeTipo, 'CAMION'),
        eq(parametrosAdaptativos.scopeId, patente),
        eq(parametrosAdaptativos.parametro,
           'fuel_level_antes_carga')
      )
    )
    .limit(1)

  if (param.length > 0 &&
      param[0].totalMuestras >= 5) {
    // Si normalmente carga con 25% del estanque
    // y hoy cargó con 70% → anticipada
    // Umbral = promedio + 2 desviaciones
    return param[0].umbralAnomalia
  }

  // Fallback: parámetro del contrato
  const paramContrato = await db.select()
    .from(parametrosAdaptativos)
    .where(
      and(
        eq(parametrosAdaptativos.scopeTipo, 'CONTRATO'),
        eq(parametrosAdaptativos.scopeId, contrato),
        eq(parametrosAdaptativos.parametro,
           'fuel_level_antes_carga')
      )
    )
    .limit(1)

  if (paramContrato.length > 0 &&
      paramContrato[0].totalMuestras >= 10) {
    return paramContrato[0].umbralAnomalia
  }

  // Sin datos → no evaluar
  return null
}
