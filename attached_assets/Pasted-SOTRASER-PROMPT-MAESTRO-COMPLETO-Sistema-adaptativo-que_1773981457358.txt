const JOBS = {

  // Cada 90 segundos — GPS en tiempo real
  GPS_SYNC: {
    intervalo: 90 * 1000,
    nombre: "GPS_SYNC",
    fn: async () => {
      await syncVolvoGPS()
    }
  },

  // Cada 5 minutos — procesar viajes nuevos
  VIAJES_NUEVOS: {
    intervalo: 5 * 60 * 1000,
    nombre: "VIAJES_NUEVOS",
    fn: async () => {
      await procesarViajesNuevos()
    }
  },

  // Cada 30 minutos — recalcular parámetros adaptativos
  PARAMETROS: {
    intervalo: 30 * 60 * 1000,
    nombre: "PARAMETROS",
    fn: async () => {
      await calcularParametros()
    }
  },

  // Cada 30 minutos — detectar cambios de patrón
  PATRONES: {
    intervalo: 30 * 60 * 1000,
    nombre: "PATRONES",
    fn: async () => {
      await detectarCambiosPatron()
    }
  },

  // Cada 6 horas — sync Sigetra para conductores
  SIGETRA_SYNC: {
    intervalo: 6 * 60 * 60 * 1000,
    nombre: "SIGETRA_SYNC",
    fn: async () => {
      const desde = new Date("2026-03-01")
      const hasta = new Date()
      await syncSigetraToCargas(desde, hasta)
    }
  },

  // Cada 24 horas a las 06:00 — reporte diario
  REPORTE_DIARIO: {
    intervalo: null, // cron, no intervalo fijo
    nombre: "REPORTE_DIARIO",
    fn: async () => {
      await generarReporteDiario()
    }
  }
}

// ─── RUNNER DE JOBS ───────────────────────────

async function runJob(job: typeof JOBS[keyof typeof JOBS]) {
  const inicio = Date.now()
  try {
    console.log(`[JOB] Iniciando ${job.nombre}`)
    await job.fn()
    const ms = Date.now() - inicio
    console.log(`[JOB] ${job.nombre} completado en ${ms}ms`)
  } catch (error: any) {
    console.error(`[JOB] ${job.nombre} falló:`, error.message)
    // NO lanzar el error — el sistema sigue corriendo
  }
}

export function iniciarJobs() {
  console.log("[JOBS] Iniciando sistema de jobs en background...")

  // GPS cada 90 segundos
  // Primer run después de 10 segundos del inicio
  setTimeout(() => {
    runJob(JOBS.GPS_SYNC)
    setInterval(() => runJob(JOBS.GPS_SYNC), JOBS.GPS_SYNC.intervalo)
  }, 10 * 1000)

  // Viajes nuevos cada 5 minutos
  setTimeout(() => {
    runJob(JOBS.VIAJES_NUEVOS)
    setInterval(() => runJob(JOBS.VIAJES_NUEVOS), JOBS.VIAJES_NUEVOS.intervalo)
  }, 30 * 1000)

  // Parámetros adaptativos cada 30 minutos
  setTimeout(() => {
    runJob(JOBS.PARAMETROS)
    setInterval(() => runJob(JOBS.PARAMETROS), JOBS.PARAMETROS.intervalo)
  }, 2 * 60 * 1000)

  // Cambios de patrón cada 30 minutos (offset 15min)
  setTimeout(() => {
    runJob(JOBS.PATRONES)
    setInterval(() => runJob(JOBS.PATRONES), JOBS.PATRONES.intervalo)
  }, 15 * 60 * 1000)

  // Sigetra cada 6 horas
  setTimeout(() => {
    runJob(JOBS.SIGETRA_SYNC)
    setInterval(() => runJob(JOBS.SIGETRA_SYNC), JOBS.SIGETRA_SYNC.intervalo)
  }, 60 * 1000)

  // Reporte diario a las 06:00
  programarReporteDiario()

  console.log("[JOBS] Todos los jobs iniciados ✅")
}

function programarReporteDiario() {
  const ahora = new Date()
  const proximas6am = new Date()
  proximas6am.setHours(6, 0, 0, 0)
  if (proximas6am <= ahora) {
    proximas6am.setDate(proximas6am.getDate() + 1)
  }
  const msHasta6am = proximas6am.getTime() - ahora.getTime()
  
  setTimeout(() => {
    runJob(JOBS.REPORTE_DIARIO)
    // Después correr cada 24h
    setInterval(() => runJob(JOBS.REPORTE_DIARIO), 24 * 60 * 60 * 1000)
  }, msHasta6am)
  
  console.log(`[JOBS] Reporte diario programado para las 06:00 (en ${Math.round(msHasta6am/60000)} min)`)
}

  console.log(`  Viajes totales procesados: ${estado.total_viajes_procesados}`)
  console.log(`  Confianza global: ${estado.confianza_global}`)
}

export async function procesarViajesNuevos() {
  
  const viajes = await db.select()
    .from(viajesAprendizaje)
    .where(eq(viajesAprendizaje.procesadoAprendizaje, false))
    .limit(50) // procesar de a 50 por ciclo
  
  if (viajes.length === 0) return
  
  console.log(`[APRENDIZAJE] Procesando ${viajes.length} viajes nuevos`)
  
  for (const viaje of viajes) {
    try {
      
      // 1. Asignar corredor
      await asignarCorredor(viaje)
      
      // 2. Actualizar perfil conductor
      if (viaje.conductor) {
        await actualizarPerfilConductor(viaje)
      }
      
      // 3. Actualizar perfil camión
      await actualizarPerfilCamion(viaje)
      
      // 4. Evaluar viaje con parámetros adaptativos
      const evaluacion = await evaluarViaje(viaje)
      
      // 5. Actualizar estado del viaje
      await db.update(viajesAprendizaje)
        .set({
          procesadoAprendizaje: true,
          validacionEstado: evaluacion.estado,
          scoreAnomalia: evaluacion.score
        })
        .where(eq(viajesAprendizaje.id, viaje.id))
      
    } catch (err: any) {
      console.error(`[APRENDIZAJE] Error en viaje ${viaje.id}:`, err.message)
      // Continuar con el siguiente
    }
  }
}

async function asignarCorredor(viaje: any) {
  // Buscar corredor existente con origen/destino similares
  const corredoresExistentes = await db.select()
    .from(corredores)
    .where(eq(corredores.activo, true))
  
  let corredorMatch = null
  
  for (const c of corredoresExistentes) {
    const distOrigen = haversineKm(
      viaje.origenLat, viaje.origenLng,
      parseFloat(c.origenLat), parseFloat(c.origenLng)
    )
    const distDestino = haversineKm(
      viaje.destinoLat, viaje.destinoLng,
      parseFloat(c.destinoLat), parseFloat(c.destinoLng)
    )
    
    if (distOrigen < 8 && distDestino < 8) {
      corredorMatch = c
      break
    }
  }
  
  if (corredorMatch) {
    // Actualizar estadísticas del corredor
    const viajesDelCorredor = await db.select()
      .from(viajesAprendizaje)
      .where(eq(viajesAprendizaje.corredorId, corredorMatch.id))
    
    const rendimientos = viajesDelCorredor
      .map(v => v.rendimientoReal)
      .filter(r => r && r > 0 && r < 15) as number[]
    
    await db.update(corredores)
      .set({
        totalViajesBase: viajesDelCorredor.length,
        rendimientoPromedio: media(rendimientos),
        rendimientoDesviacion: stdDev(rendimientos),
        updatedAt: new Date()
      })
      .where(eq(corredores.id, corredorMatch.id))
    
    // Asignar corredor al viaje
    await db.update(viajesAprendizaje)
      .set({ corredorId: corredorMatch.id })
      .where(eq(viajesAprendizaje.id, viaje.id))
      
  } else {
    // Crear nuevo corredor
    const nombreOrigen = await geocodearLugar(
      viaje.origenLat, viaje.origenLng
    )
    const nombreDestino = await geocodearLugar(
      viaje.destinoLat, viaje.destinoLng
    )
    
    const nuevoCorredor = await db.insert(corredores)
      .values({
        nombre: `${nombreOrigen} → ${nombreDestino}`,
        contrato: viaje.contrato,
        origenNombre: nombreOrigen,
        destinoNombre: nombreDestino,
        origenLat: String(viaje.origenLat),
        origenLng: String(viaje.origenLng),
        destinoLat: String(viaje.destinoLat),
        destinoLng: String(viaje.destinoLng),
        radioToleranciaKm: 8,
        totalViajesBase: 1,
        activo: true,
        updatedAt: new Date()
      })
      .returning()
    
    await db.update(viajesAprendizaje)
      .set({ corredorId: nuevoCorredor[0].id })
      .where(eq(viajesAprendizaje.id, viaje.id))
  }
}

async function actualizarPerfilConductor(viaje: any) {
  const existente = await db.select()
    .from(conductoresPerfil)
    .where(eq(conductoresPerfil.nombre, viaje.conductor))
    .limit(1)
  
  if (existente.length > 0) {
    const p = existente[0]
    const nuevoTotal = p.totalJornadas + 1
    const nuevoKm = p.kmTotal + (viaje.kmGps || 0)
    const nuevoLitros = p.litrosTotalEcu + (viaje.litrosEcu || 0)
    const nuevoRend = nuevoKm > 0 && nuevoLitros > 0 
      ? nuevoKm / nuevoLitros : p.rendimientoPromedio
    
    await db.update(conductoresPerfil)
      .set({
        totalJornadas: nuevoTotal,
        kmTotal: nuevoKm,
        litrosTotalEcu: nuevoLitros,
        rendimientoPromedio: nuevoRend,
        ultimaJornada: new Date(viaje.fechaInicio),
        contrato: viaje.contrato
      })
      .where(eq(conductoresPerfil.id, p.id))
  } else {
    await db.insert(conductoresPerfil).values({
      nombre: viaje.conductor,
      contrato: viaje.contrato,
      totalJornadas: 1,
      kmTotal: viaje.kmGps || 0,
      litrosTotalEcu: viaje.litrosEcu || 0,
      rendimientoPromedio: viaje.rendimientoReal || null,
      ultimaJornada: new Date(viaje.fechaInicio),
      primeraJornada: new Date(viaje.fechaInicio)
    })
  }
}

async function evaluarViaje(viaje: any) {
  // Buscar parámetros del corredor
  const params = await db.select()
    .from(parametrosAdaptativos)
    .where(
      and(
        eq(parametrosAdaptativos.scopeTipo, 'CORREDOR'),
        eq(parametrosAdaptativos.scopeId, String(viaje.corredorId)),
        eq(parametrosAdaptativos.parametro, 'rendimiento_kmL')
      )
    )
    .limit(1)
  
  if (!params.length || params[0].totalMuestras < 3) {
    return { estado: 'PENDIENTE', score: 0 }
  }
  
  const p = params[0]
  const rend = viaje.rendimientoReal || 0
  const zScore = (rend - p.valorPromedio) / p.valorDesviacion
  
  if (zScore >= -1.0) return { estado: 'NORMAL', score: 0 }
  if (zScore >= -1.5) return { estado: 'REVISAR', score: 25 }
  if (zScore >= -2.0) return { estado: 'ANOMALIA', score: 60 }
  return { estado: 'CRITICO', score: 90 }
}

export async function calcularParametros() {
  console.log('[PARAMETROS] Recalculando parámetros adaptativos...')
  
  const todosCorredores = await db.select().from(corredores)
  
  for (const corredor of todosCorredores) {
    const viajes = await db.select()
      .from(viajesAprendizaje)
      .where(eq(viajesAprendizaje.corredorId, corredor.id))
    
    if (viajes.length < 3) continue
    
    const rendimientos = viajes
      .map(v => v.rendimientoReal)
      .filter(r => r && r > 0 && r < 15) as number[]
    
    if (rendimientos.length < 3) continue
    
    const prom = media(rendimientos)
    const desv = stdDev(rendimientos)
    const confianza = 
      rendimientos.length >= 200 ? 'EXPERTA' :
      rendimientos.length >= 50  ? 'ALTA' :
      rendimientos.length >= 10  ? 'MEDIA' : 'BAJA'
    
    // Upsert del parámetro
    await db.execute(sql`
      INSERT INTO parametros_adaptativos 
        (scope_tipo, scope_id, parametro, valor_promedio, 
         valor_desviacion, umbral_revisar, umbral_anomalia,
         umbral_critico, total_muestras, confianza,
         ultima_actualizacion)
      VALUES 
        ('CORREDOR', ${String(corredor.id)}, 'rendimiento_kmL',
         ${prom}, ${desv}, ${prom - 1.5*desv}, 
         ${prom - 2.0*desv}, ${prom - 3.0*desv},
         ${rendimientos.length}, ${confianza}, now())
      ON CONFLICT (scope_tipo, scope_id, parametro)
      DO UPDATE SET
        valor_promedio = EXCLUDED.valor_promedio,
        valor_desviacion = EXCLUDED.valor_desviacion,
        umbral_revisar = EXCLUDED.umbral_revisar,
        umbral_anomalia = EXCLUDED.umbral_anomalia,
        umbral_critico = EXCLUDED.umbral_critico,
        total_muestras = EXCLUDED.total_muestras,
        confianza = EXCLUDED.confianza,
        ultima_actualizacion = now()
    `)
  }
  
  console.log('[PARAMETROS] Recálculo completado')
}

export async function detectarCambiosPatron() {
  console.log('[PATRONES] Detectando cambios de patrón...')
  
  const conductores = await db.select().from(conductoresPerfil)
  
  for (const conductor of conductores) {
    if (conductor.totalJornadas < 5) continue
    
    // Últimos 3 viajes del conductor
    const ultimos = await db.select()
      .from(viajesAprendizaje)
      .where(eq(viajesAprendizaje.conductor, conductor.nombre))
      .orderBy(desc(viajesAprendizaje.fechaInicio))
      .limit(3)
    
    if (ultimos.length < 3) continue
    
    const rendReciente = media(
      ultimos.map(v => v.rendimientoReal || 0)
        .filter(r => r > 0)
    )
    
    if (!conductor.rendimientoPromedio || rendReciente === 0) continue
    
    const diferenciaPct = ((rendReciente - conductor.rendimientoPromedio) 
      / conductor.rendimientoPromedio) * 100
    
    // Si bajó más del 20% vs su historial
    if (diferenciaPct < -20) {
      // Verificar que no existe alerta similar reciente
      const alertaExistente = await db.select()
        .from(alertasAprendizaje)
        .where(
          and(
            eq(alertasAprendizaje.entidadNombre, conductor.nombre),
            eq(alertasAprendizaje.tipo, 'CAMBIO_PATRON_CONDUCTOR'),
            eq(alertasAprendizaje.gestionado, false)
          )
        )
        .limit(1)
      
      if (alertaExistente.length === 0) {
        await db.insert(alertasAprendizaje).values({
          tipo: 'CAMBIO_PATRON_CONDUCTOR',
          entidadTipo: 'conductor',
          entidadNombre: conductor.nombre,
          contrato: conductor.contrato || '',
          descripcion: `Rendimiento cayó ${Math.abs(diferenciaPct).toFixed(0)}% ` +
            `en últimos 3 viajes vs historial. ` +
            `Reciente: ${rendReciente.toFixed(2)} km/L, ` +
            `Histórico: ${conductor.rendimientoPromedio.toFixed(2)} km/L`,
          valorReciente: rendReciente,
          valorHistorico: conductor.rendimientoPromedio,
          diferenciaPct
        })
        
        console.log(`[PATRONES] Alerta creada: ${conductor.nombre} bajó ${Math.abs(diferenciaPct).toFixed(0)}%`)
      }
    }
  }
}

// Helpers
function haversineKm(lat1: number, lng1: number, lat2: number, lng2: number): number {
  const R = 6371
  const dLat = (lat2 - lat1) * Math.PI / 180
  const dLng = (lng2 - lng1) * Math.PI / 180
  const a = Math.sin(dLat/2)**2 + 
    Math.cos(lat1*Math.PI/180) * Math.cos(lat2*Math.PI/180) * Math.sin(dLng/2)**2
  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a))
}

function media(vals: number[]): number {
  if (vals.length === 0) return 0
  return vals.reduce((a,b) => a+b, 0) / vals.length
}

function stdDev(vals: number[]): number {
  if (vals.length < 2) return 0
  const m = media(vals)
  return Math.sqrt(vals.reduce((s,v) => s + (v-m)**2, 0) / vals.length)
}

async function geocodearLugar(lat: number, lng: number): Promise<string> {
  try {
    const lugares = await db.select().from(geoLugares)
    let mejorLugar = 'Punto desconocido'
    let menorDist = 999
    
    for (const lugar of lugares) {
      const dist = haversineKm(lat, lng, 
        parseFloat(lugar.lat), parseFloat(lugar.lng))
      if (dist < menorDist && dist < 15) {
        menorDist = dist
        mejorLugar = lugar.nombre
      }
    }
    return mejorLugar
  } catch {
    return 'Punto desconocido'
  }
}

export async function getSistemaEstado() {
  
  const [
    primerSnapshot,
    totalViajes,
    totalCorredores,
    totalConductores,
    totalCamiones,
    totalSnapshots,
    alertasActivas,
    parametrosAlta
  ] = await Promise.all([
    db.select().from(volvoFuelSnapshots)
      .orderBy(asc(volvoFuelSnapshots.capturedAt)).limit(1),
    db.select({ count: count() }).from(viajesAprendizaje),
    db.select({ count: count() }).from(corredores)
      .where(gte(corredores.totalViajesBase, 3)),
    db.select({ count: count() }).from(conductoresPerfil),
    db.select({ count: count() }).from(camionesPerfil),
    db.select({ count: count() }).from(volvoFuelSnapshots),
    db.select({ count: count() }).from(alertasAprendizaje)
      .where(eq(alertasAprendizaje.gestionado, false)),
    db.select({ count: count() }).from(parametrosAdaptativos)
      .where(inArray(parametrosAdaptativos.confianza, ['ALTA', 'EXPERTA']))
  ])
  
  const diasAprendiendo = primerSnapshot.length > 0
    ? Math.floor((Date.now() - 
        new Date(primerSnapshot[0].capturedAt).getTime()) 
        / (1000 * 60 * 60 * 24))
    : 0
  
  const nViajes = totalViajes[0].count
  const confianzaGlobal = 
    nViajes >= 1000 ? 'EXPERTA' :
    nViajes >= 200  ? 'ALTA' :
    nViajes >= 50   ? 'MEDIA' : 'BAJA'
  
  const madurez = Math.min(100, Math.floor(
    Math.min(diasAprendiendo * 2, 40) +
    Math.min(nViajes / 10, 40) +
    Math.min(totalCorredores[0].count * 2, 20)
  ))
  
  return {
    dias_aprendiendo: diasAprendiendo,
    total_viajes_procesados: nViajes,
    total_corredores_conocidos: totalCorredores[0].count,
    total_conductores_analizados: totalConductores[0].count,
    total_camiones_perfilados: totalCamiones[0].count,
    total_snapshots_volvo: totalSnapshots[0].count,
    alertas_patron_activas: alertasActivas[0].count,
    parametros_calibrados: parametrosAlta[0].count,
    confianza_global: confianzaGlobal,
    madurez_pct: madurez,
    primer_dato: primerSnapshot[0]?.capturedAt || null,
    estado_mensaje: generarMensaje(diasAprendiendo, nViajes, confianzaGlobal),
    timestamp: new Date().toISOString()
  }
}

function generarMensaje(dias: number, viajes: number, confianza: string): string {
  if (dias === 0) return "Sistema iniciando — recopilando primeros datos de Volvo Connect"
  if (confianza === 'BAJA') return `${dias} días activo con ${viajes} viajes — acumulando datos para calibrar parámetros`
  if (confianza === 'MEDIA') return `${dias} días activo — ${viajes} viajes procesados, parámetros tomando forma por contrato`
  if (confianza === 'ALTA') return `Sistema calibrado — ${viajes} viajes analizados en ${dias} días, detección confiable activa`
  return `Sistema experto — modelo altamente preciso con ${viajes} viajes en ${dias} días`
}
