PARTE 6 — BACKEND ENDPOINT RENDIMIENTO
═══════════════════════════════════════════════

Agregar en server/geo-routes.ts:

app.get("/api/geo/rendimiento-contratos", 
async (req, res) => {
  try {
    const camiones = await storage.getCamiones()
    const faenas = await storage.getFaenas()
    
    const resultado = []
    
    for (const faena of faenas) {
      const camionesContrato = camiones.filter(
        c => c.faenaId === faena.id && c.vin != null
      )
      if (camionesContrato.length === 0) continue
      
      const vins = camionesContrato
        .map(c => c.vin)
        .filter(Boolean)
      
      // Snapshots últimos 7 días
      const desde = new Date(Date.now() - 7*24*60*60*1000)
      const hasta = new Date()
      
      const snapshots = await storage
        .getVolvoFuelSnapshotsInRange(vins, desde, hasta)
      
      // Calcular por camión
      const porCamion = []
      for (const cam of camionesContrato) {
        if (!cam.vin) continue
        const snaps = snapshots
          .filter(s => s.vin === cam.vin)
          .sort((a,b) => 
            new Date(a.capturedAt).getTime() - 
            new Date(b.capturedAt).getTime()
          )
        if (snaps.length < 2) continue
        
        const deltaFuel = 
          snaps[snaps.length-1].totalFuelUsed - 
          snaps[0].totalFuelUsed
        const deltaKm = 
          (snaps[snaps.length-1].totalDistance || 0) - 
          (snaps[0].totalDistance || 0)
        
        const litros = deltaFuel / 1000
        const km = deltaKm / 1000
        const rend = litros > 0 ? km / litros : null
        
        porCamion.push({
          patente: cam.patente,
          meta: cam.metaKmL,
          rendimiento_real: rend 
            ? Math.round(rend * 100) / 100 
            : null,
          bajo_meta: rend != null && rend < cam.metaKmL * 0.7,
          diferencia_pct: rend 
            ? Math.round(((rend - cam.metaKmL) / cam.metaKmL) * 100) 
            : null
        })
      }
      
      if (porCamion.length === 0) continue
      
      const conRend = porCamion
        .filter(c => c.rendimiento_real != null)
      const rendProm = conRend.length > 0
        ? conRend.reduce((s,c) => s + c.rendimiento_real!, 0) 
          / conRend.length
        : null
      const metaProm = camionesContrato
        .reduce((s,c) => s + c.metaKmL, 0) 
        / camionesContrato.length
      
      resultado.push({
        nombre: faena.nombre,
        color: faena.color,
        camiones_total: camionesContrato.length,
        camiones_con_datos: conRend.length,
        rendimiento_promedio: rendProm 
          ? Math.round(rendProm * 100) / 100 
          : null,
        meta_kmL: Math.round(metaProm * 100) / 100,
        bajo_meta: porCamion.filter(c => c.bajo_meta)
      })
    }
    
    res.json(resultado)
  } catch (error: any) {
    console.error("[GEO-V2] rendimiento-contratos:", error)
    res.status(500).json({ message: error.message })
  }
})
