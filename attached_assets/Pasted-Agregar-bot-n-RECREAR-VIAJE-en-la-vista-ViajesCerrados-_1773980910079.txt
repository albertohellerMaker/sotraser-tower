function ReplayModal({ 
  viaje, 
  onClose 
}: { 
  viaje: any, 
  onClose: () => void 
}) {
  
  // Estados
  const mapRef = useRef<HTMLDivElement>(null)
  const mapInstance = useRef<any>(null)
  const [progreso, setProgreso] = useState(0) // 0-100
  const [reproduciendo, setReproduciendo] = useState(false)
  const [terminado, setTerminado] = useState(false)
  const [puntosGps, setPuntosGps] = useState<any[]>([])
  const [cargando, setCargando] = useState(true)
  const polylineRef = useRef<any>(null)
  const marcadorRef = useRef<any>(null)
  const intervalRef = useRef<any>(null)

  // 1. Al montar: cargar puntos GPS del viaje
  useEffect(() => {
    fetch(`/api/geo/viaje-puntos/${viaje.id}`)
      .then(r => r.json())
      .then(data => {
        setPuntosGps(data.puntos || [])
        setCargando(false)
      })
      .catch(() => setCargando(false))
  }, [viaje.id])

  // 2. Inicializar mapa Leaflet
  useEffect(() => {
    if (cargando || !mapRef.current || mapInstance.current) return
    
    const L = (window as any).L
    if (!L) return
    
    // Centrar mapa en el viaje
    const centerLat = viaje.origenLat || -33.45
    const centerLng = viaje.origenLng || -70.65
    
    const map = L.map(mapRef.current).setView(
      [centerLat, centerLng], 7
    )
    L.tileLayer(
      "https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png",
      { attribution: "© OpenStreetMap" }
    ).addTo(map)
    
    mapInstance.current = map
    
    // Dibujar marcadores de origen y destino
    if (viaje.origenLat && viaje.origenLng) {
      L.circleMarker([viaje.origenLat, viaje.origenLng], {
        radius: 8, 
        fillColor: "#00ff88", 
        color: "#020508",
        weight: 2,
        fillOpacity: 1
      }).addTo(map)
       .bindPopup(`<b>ORIGEN</b><br>${viaje.origenNombre || "Inicio"}`)
    }
    
    if (viaje.destinoLat && viaje.destinoLng) {
      L.circleMarker([viaje.destinoLat, viaje.destinoLng], {
        radius: 8,
        fillColor: "#ff2244",
        color: "#020508", 
        weight: 2,
        fillOpacity: 1
      }).addTo(map)
       .bindPopup(`<b>DESTINO</b><br>${viaje.destinoNombre || "Fin"}`)
    }
    
    // Si hay puntos GPS, ajustar bounds del mapa
    if (puntosGps.length > 0) {
      const bounds = puntosGps
        .filter(p => p.lat && p.lng)
        .map(p => [p.lat, p.lng])
      if (bounds.length > 1) {
        map.fitBounds(bounds, { padding: [40, 40] })
      }
    }
    
  }, [cargando, puntosGps])

  // 3. Lógica de animación en 7 segundos
  const iniciarReplay = () => {
    const L = (window as any).L
    if (!L || !mapInstance.current) return
    if (puntosGps.length < 2) return
    
    setReproduciendo(true)
    setTerminado(false)
    setProgreso(0)
    
    // Limpiar polyline anterior
    if (polylineRef.current) {
      polylineRef.current.remove()
    }
    if (marcadorRef.current) {
      marcadorRef.current.remove()
    }
    
    // Crear ícono del camión animado
    const iconCamion = L.divIcon({
      html: `<div style="
        width:20px;height:20px;
        background:#00d4ff;
        border:2px solid #020508;
        border-radius:50%;
        box-shadow:0 0 8px #00d4ff;
      "></div>`,
      className: "",
      iconSize: [20, 20],
      iconAnchor: [10, 10]
    })
    
    const puntosFiltrados = puntosGps.filter(p => p.lat && p.lng)
    const total = puntosFiltrados.length
    const DURACION_MS = 7000 // 7 segundos exactos
    const intervaloMs = DURACION_MS / total
    
    let idx = 0
    const coordsAnimadas: [number, number][] = []
    
    // Colocar marcador en el inicio
    marcadorRef.current = L.marker(
      [puntosFiltrados[0].lat, puntosFiltrados[0].lng],
      { icon: iconCamion }
    ).addTo(mapInstance.current)
    
    intervalRef.current = setInterval(() => {
      if (idx >= total) {
        clearInterval(intervalRef.current)
        setReproduciendo(false)
        setTerminado(true)
        setProgreso(100)
        return
      }
      
      const punto = puntosFiltrados[idx]
      coordsAnimadas.push([punto.lat, punto.lng])
      
      // Actualizar polyline
      if (polylineRef.current) {
        polylineRef.current.remove()
      }
      polylineRef.current = L.polyline(coordsAnimadas, {
        color: "#00d4ff",
        weight: 3,
        opacity: 0.8
      }).addTo(mapInstance.current)
      
      // Mover marcador
      if (marcadorRef.current) {
        marcadorRef.current.setLatLng([punto.lat, punto.lng])
      }
      
      // Actualizar progreso
      setProgreso(Math.round((idx / total) * 100))
      
      // Seguir al camión en el mapa
      if (idx % 5 === 0) {
        mapInstance.current.panTo(
          [punto.lat, punto.lng],
          { animate: true, duration: 0.5 }
        )
      }
      
      idx++
    }, intervaloMs)
  }

  // Cleanup al desmontar
  useEffect(() => {
    return () => {
      if (intervalRef.current) clearInterval(intervalRef.current)
    }
  }, [])

  return (
    <div 
      className="fixed inset-0 z-[60] flex items-center 
                 justify-center bg-black/85"
      onClick={onClose}
    >
      <div
        className="w-full max-w-4xl mx-4 flex flex-col"
        style={{
          background: "#020508",
          border: "1px solid rgba(0,212,255,0.3)",
          maxHeight: "90vh"
        }}
        onClick={e => e.stopPropagation()}
      >
        {/* Header */}
        <div 
          className="flex items-center justify-between p-4 
                     flex-shrink-0"
          style={{ borderBottom: "1px solid #0d2035" }}
        >
          <div className="flex items-center gap-3">
            <div 
              className="font-space text-sm font-bold 
                         tracking-[0.15em]"
              style={{ color: "#00d4ff" }}
            >
              ▶ RECREAR VIAJE
            </div>
            <div 
              className="font-rajdhani text-sm"
              style={{ color: "#3a6080" }}
            >
              {viaje.patente} · {viaje.origenNombre || "Origen"} 
              → {viaje.destinoNombre || "Destino"}
            </div>
          </div>
          <div className="flex items-center gap-3">
            <div 
              className="font-exo text-[10px]"
              style={{ color: "#3a6080" }}
            >
              {puntosGps.length} puntos GPS · 
              {viaje.kmRecorridos 
                ? ` ${Math.round(viaje.kmRecorridos)} km` 
                : ""}
            </div>
            <button 
              onClick={onClose}
              className="p-1 cursor-pointer hover:opacity-70"
            >
              <X className="w-4 h-4" style={{ color: "#3a6080" }} />
            </button>
          </div>
        </div>

        {/* Mapa */}
        <div className="flex-1 relative" style={{ minHeight: "400px" }}>
          {cargando && (
            <div 
              className="absolute inset-0 flex flex-col 
                         items-center justify-center z-10"
              style={{ background: "#020508" }}
            >
              <Loader2 
                className="w-8 h-8 animate-spin mb-3"
                style={{ color: "#00d4ff" }}
              />
              <span 
                className="font-exo text-sm"
                style={{ color: "#3a6080" }}
              >
                Cargando puntos GPS del viaje...
              </span>
            </div>
          )}
          
          {!cargando && puntosGps.length === 0 && (
            <div 
              className="absolute inset-0 flex flex-col 
                         items-center justify-center"
            >
              <MapPin 
                className="w-8 h-8 mb-3"
                style={{ color: "#3a6080" }}
              />
              <span 
                className="font-exo text-sm"
                style={{ color: "#3a6080" }}
              >
                Sin puntos GPS para este viaje
              </span>
              <span 
                className="font-exo text-[10px] mt-1"
                style={{ color: "#1a3a55" }}
              >
                El viaje fue reconstruido sin trayectoria GPS detallada
              </span>
            </div>
          )}
          
          <div ref={mapRef} style={{ width: "100%", height: "100%" }} />
        </div>

        {/* Controles de replay */}
        <div 
          className="p-4 flex-shrink-0"
          style={{ borderTop: "1px solid #0d2035" }}
        >
          {/* Barra de progreso */}
          <div 
            className="h-1 w-full mb-3 overflow-hidden"
            style={{ background: "#0d2035" }}
          >
            <div
              className="h-full transition-all"
              style={{
                width: `${progreso}%`,
                background: terminado ? "#00ff88" : "#00d4ff",
                boxShadow: reproduciendo 
                  ? "0 0 8px rgba(0,212,255,0.6)" 
                  : "none",
                transition: "width 0.1s linear"
              }}
            />
          </div>

          <div className="flex items-center justify-between">
            {/* Info del viaje */}
            <div className="flex items-center gap-4">
              <div>
                <div 
                  className="font-exo text-[9px] tracking-[0.15em]"
                  style={{ color: "#3a6080" }}
                >
                  DURACIÓN
                </div>
                <div 
                  className="font-space text-sm font-bold"
                  style={{ color: "#c8e8ff" }}
                >
                  {viaje.fechaSalida && viaje.fechaLlegada
                    ? (() => {
                        const ms = new Date(viaje.fechaLlegada).getTime() 
                          - new Date(viaje.fechaSalida).getTime()
                        const h = Math.floor(ms / 3600000)
                        const m = Math.floor((ms % 3600000) / 60000)
                        return `${h}h ${m}m`
                      })()
                    : "--"
                  }
                </div>
              </div>
              <div>
                <div 
                  className="font-exo text-[9px] tracking-[0.15em]"
                  style={{ color: "#3a6080" }}
                >
                  RENDIMIENTO
                </div>
                <div 
                  className="font-space text-sm font-bold"
                  style={{ 
                    color: viaje.rendimiento >= 2.5 
                      ? "#00ff88" 
                      : "#ff2244" 
                  }}
                >
                  {viaje.rendimiento 
                    ? `${viaje.rendimiento.toFixed(2)} km/L` 
                    : "--"
                  }
                </div>
              </div>
              <div>
                <div 
                  className="font-exo text-[9px] tracking-[0.15em]"
                  style={{ color: "#3a6080" }}
                >
                  VELOCIDAD MÁX
                </div>
                <div 
                  className="font-space text-sm font-bold"
                  style={{ 
                    color: (viaje.velocidadMaxima || 0) > 100 
                      ? "#ff2244" 
                      : "#c8e8ff" 
                  }}
                >
                  {viaje.velocidadMaxima 
                    ? `${Math.round(viaje.velocidadMaxima)} km/h` 
                    : "--"
                  }
                </div>
              </div>
            </div>

            {/* Botones de control */}
            <div className="flex items-center gap-2">
              {!reproduciendo && !terminado && 
               puntosGps.length > 0 && (
                <button
                  onClick={iniciarReplay}
                  className="flex items-center gap-2 px-5 py-2.5 
                             font-space text-[11px] font-bold 
                             cursor-pointer transition-all 
                             hover:opacity-80"
                  style={{
                    background: "rgba(0,212,255,0.1)",
                    border: "1px solid rgba(0,212,255,0.4)",
                    color: "#00d4ff",
                    boxShadow: "0 0 12px rgba(0,212,255,0.15)"
                  }}
                >
                  <Play className="w-4 h-4" />
                  INICIAR RECREACIÓN · 7s
                </button>
              )}
              
              {reproduciendo && (
                <div className="flex items-center gap-2">
                  <div 
                    className="w-2 h-2 rounded-full"
                    style={{ 
                      background: "#00d4ff",
                      animation: "blink 0.5s infinite"
                    }}
                  />
                  <span 
                    className="font-space text-[11px] font-bold"
                    style={{ color: "#00d4ff" }}
                  >
                    REPRODUCIENDO... {progreso}%
                  </span>
                </div>
              )}
              
              {terminado && (
                <button
                  onClick={iniciarReplay}
                  className="flex items-center gap-2 px-4 py-2 
                             font-space text-[11px] font-bold 
                             cursor-pointer hover:opacity-80"
                  style={{
                    background: "rgba(0,255,136,0.08)",
                    border: "1px solid rgba(0,255,136,0.3)",
                    color: "#00ff88"
                  }}
                >
                  <RotateCcw className="w-3.5 h-3.5" />
                  REPETIR
                </button>
              )}
            </div>
          </div>
        </div>
      </div>
    </div>
  )
}

GET /api/geo/viaje-puntos/:viajeId

  Buscar el viaje en geo_viajes o viajes_aprendizaje
  
  Opción A — Si el viaje tiene trayectoria en geo_puntos:
    SELECT lat, lng, velocidad_kmh as velocidad,
           timestamp_punto as timestamp,
           km_odometro
    FROM geo_puntos
    WHERE patente = [patente del viaje]
      AND timestamp_punto >= [fecha_salida del viaje]
      AND timestamp_punto <= [fecha_llegada del viaje]
    ORDER BY timestamp_punto ASC
  
  Opción B — Si no hay puntos GPS:
    Retornar array vacío con mensaje explicativo
  
  Respuesta:
  {
    viaje_id: number,
    patente: string,
    total_puntos: number,
    tiene_gps: boolean,
    puntos: [
      {
        lat: number,
        lng: number,
        velocidad: number,
        timestamp: string,
        km_odometro: number
      }
    ]
  }
