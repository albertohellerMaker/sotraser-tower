Retorna el estado del aprendizaje de patrones:

async function getAprendizajeEstaciones(req, res) {
  
  // Total patrones aprendidos por nivel
  const patronesPorConfianza = await db.select({
    confianza: patronesCargaCombustible.confianza,
    count: count()
  })
  .from(patronesCargaCombustible)
  .where(eq(patronesCargaCombustible.activo, true))
  .groupBy(patronesCargaCombustible.confianza)
  
  // Total camiones con patrón propio
  const camionesConPatron = await db.select({
    count: count()
  })
  .from(patronesCargaCombustible)
  .where(
    and(
      eq(patronesCargaCombustible.scopeTipo, 'CAMION'),
      eq(patronesCargaCombustible.activo, true)
    )
  )
  
  // Total estaciones conocidas con patrón
  const estacionesConocidas = await db.select({
    count: count()
  })
  .from(patronesCargaCombustible)
  .where(
    and(
      eq(patronesCargaCombustible.scopeTipo, 'CAMION_ESTACION'),
      eq(patronesCargaCombustible.activo, true)
    )
  )
  
  // Total cargas analizadas históricamente
  const totalCargasHistoricas = await db.select({
    count: count()
  })
  .from(cargasTable)
  .where(
    and(
      isNotNull(cargasTable.litros),
      gt(cargasTable.litros, 0)
    )
  )
  
  // Top 5 camiones con más cargas aprendidas
  const topCamiones = await db.select()
    .from(patronesCargaCombustible)
    .where(eq(patronesCargaCombustible.scopeTipo, 'CAMION'))
    .orderBy(desc(patronesCargaCombustible.totalCargas))
    .limit(5)
  
  // Top 5 estaciones más conocidas
  const topEstaciones = await db.select({
    estacion: patronesCargaCombustible.estacionNombre,
    total_cargas: sql`SUM(${patronesCargaCombustible.totalCargas})`,
    camiones: count()
  })
  .from(patronesCargaCombustible)
  .where(eq(patronesCargaCombustible.scopeTipo, 'CAMION_ESTACION'))
  .groupBy(patronesCargaCombustible.estacionNombre)
  .orderBy(desc(sql`SUM(${patronesCargaCombustible.totalCargas})`))
  .limit(5)
  
  // Última actualización del aprendizaje
  const ultimaActualizacion = await db.select({
    ultima: sql`MAX(${patronesCargaCombustible.ultimaActualizacion})`
  })
  .from(patronesCargaCombustible)
  
  // Calcular madurez global 0-100
  const totalPatrones = patronesPorConfianza
    .reduce((s, p) => s + p.count, 0)
  const patronesAltos = patronesPorConfianza
    .filter(p => ['ALTA', 'EXPERTA'].includes(p.confianza))
    .reduce((s, p) => s + p.count, 0)
  const madurezPct = totalPatrones > 0
    ? Math.round((patronesAltos / totalPatrones) * 100)
    : 0
