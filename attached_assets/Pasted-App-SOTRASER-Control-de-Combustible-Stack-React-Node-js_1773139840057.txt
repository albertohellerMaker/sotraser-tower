═══════════════════════════════════════════════════

CREATE TABLE IF NOT EXISTS viajes (
  id SERIAL PRIMARY KEY,
  codigo VARCHAR(20) UNIQUE NOT NULL,
  camion_id INTEGER REFERENCES camiones(id),
  conductor VARCHAR(100) NOT NULL,
  cliente VARCHAR(100),
  origen_nombre VARCHAR(200),
  origen_lat NUMERIC(10,7),
  origen_lng NUMERIC(10,7),
  fecha_salida TIMESTAMP,
  fecha_llegada_estimada TIMESTAMP,
  fecha_cierre TIMESTAMP,
  estado VARCHAR(20) DEFAULT 'PROGRAMADO',
  -- PROGRAMADO · EN_RUTA · COMPLETADO · CANCELADO
  km_inicio NUMERIC(10,1),
  km_cierre NUMERIC(10,1),
  km_recorridos NUMERIC(10,1),
  litros_cargados NUMERIC(8,2),
  litros_ecu NUMERIC(8,2),
  diferencia_litros NUMERIC(8,2),
  carga_descripcion TEXT,
  notas TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS viaje_paradas (
  id SERIAL PRIMARY KEY,
  viaje_id INTEGER REFERENCES viajes(id),
  orden INTEGER NOT NULL,
  nombre VARCHAR(200) NOT NULL,
  direccion VARCHAR(300),
  lat NUMERIC(10,7),
  lng NUMERIC(10,7),
  tipo VARCHAR(20) DEFAULT 'ENTREGA',
  -- ENTREGA · CARGA · DESCANSO · COMBUSTIBLE
  estado VARCHAR(20) DEFAULT 'PENDIENTE',
  -- PENDIENTE · EN_CAMINO · LLEGADO · COMPLETADO
  hora_estimada TIMESTAMP,
  hora_real TIMESTAMP,
  notas TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_viajes_estado 
  ON viajes(estado);
CREATE INDEX IF NOT EXISTS idx_viajes_camion 
  ON viajes(camion_id);
CREATE INDEX IF NOT EXISTS idx_paradas_viaje 
  ON viaje_paradas(viaje_id, orden);

ENDPOINTS TMS
═══════════════════════════════════════════════════

GET  /api/tms/viajes
     Query params: estado, fecha_desde, fecha_hasta
     Devuelve viajes con sus paradas y datos Volvo
     actuales del camión (GPS, km, velocidad)

GET  /api/tms/viajes/:id
     Viaje completo con paradas, historial GPS
     del camión desde Volvo y cargas Sigetra

POST /api/tms/viajes
     Crear viaje nuevo con paradas

PUT  /api/tms/viajes/:id
     Actualizar viaje (estado, datos cierre)

PUT  /api/tms/viajes/:id/parada/:parada_id
     Actualizar estado de una parada

POST /api/tms/viajes/:id/cerrar
     Cierre de viaje:
     1. Calcular km recorridos (km_cierre - km_inicio)
     2. Buscar cargas Sigetra del camión en el período
     3. Calcular litros ECU del período en volvo_ecu
     4. Calcular diferencia (cuadratura del viaje)
     5. Marcar estado COMPLETADO con fecha_cierre

GET  /api/tms/viajes/:id/posicion
     Posición GPS actual del camión desde volvo_ecu
     más reciente + progreso hacia siguiente parada
