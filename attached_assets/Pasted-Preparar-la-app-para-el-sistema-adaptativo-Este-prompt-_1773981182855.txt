export async function getSistemaEstado() {
  
  // Tiempo que lleva el sistema aprendiendo
  const primerSnapshot = await db.select()
    .from(volvoFuelSnapshots)
    .orderBy(asc(volvoFuelSnapshots.capturedAt))
    .limit(1)
  
  const primerViaje = await db.select()
    .from(viajesAprendizaje)
    .orderBy(asc(viajesAprendizaje.fechaInicio))
    .limit(1)
  
  const diasAprendiendo = primerSnapshot.length > 0
    ? Math.floor((Date.now() - new Date(primerSnapshot[0].capturedAt).getTime()) 
      / (1000 * 60 * 60 * 24))
    : 0
  
  // Conteos de aprendizaje
  const totalViajes = await db.select({ count: count() })
    .from(viajesAprendizaje)
  
  const totalCorredores = await db.select({ count: count() })
    .from(corredores)
    .where(gte(corredores.totalViajesBase, 3))
  
  const totalConductores = await db.select({ count: count() })
    .from(conductoresPerfil)
  
  const totalCamionesPerfilados = await db.select({ count: count() })
    .from(camionesPerfil)
  
  const totalSnapshots = await db.select({ count: count() })
    .from(volvoFuelSnapshots)
  
  // Nivel de confianza global
  const confianzaGlobal = 
    totalViajes[0].count >= 1000 ? 'EXPERTA' :
    totalViajes[0].count >= 200  ? 'ALTA' :
    totalViajes[0].count >= 50   ? 'MEDIA' : 'BAJA'
  
  // Puntaje de madurez 0-100
  const madurez = Math.min(100, Math.floor(
    (diasAprendiendo * 2) +           // máx 60pts por tiempo
    (totalViajes[0].count / 20) +      // máx 50pts por viajes
    (totalCorredores[0].count * 2)     // puntos por rutas conocidas
  ))
  
  // Últimas alertas de patrón
  const alertasRecientes = await db.select()
    .from(alertasAprendizaje)
    .where(eq(alertasAprendizaje.gestionado, false))
    .orderBy(desc(alertasAprendizaje.fecha))
    .limit(5)
  
  return {
    dias_aprendiendo: diasAprendiendo,
    total_viajes_procesados: totalViajes[0].count,
    total_corredores_conocidos: totalCorredores[0].count,
    total_conductores_analizados: totalConductores[0].count,
    total_camiones_perfilados: totalCamionesPerfilados[0].count,
    total_snapshots_volvo: totalSnapshots[0].count,
    confianza_global: confianzaGlobal,
    madurez_pct: madurez,
    alertas_patron_activas: alertasRecientes.length,
    alertas_recientes: alertasRecientes,
    primer_dato: primerSnapshot[0]?.capturedAt || null,
    estado_mensaje: generarMensajeEstado(diasAprendiendo, totalViajes[0].count, confianzaGlobal)
  }
}

function generarMensajeEstado(dias: number, viajes: number, confianza: string): string {
  if (dias === 0) return "Sistema iniciando — recopilando primeros datos"
  if (confianza === 'BAJA') return `${dias} días aprendiendo — acumulando datos para calibrar`
  if (confianza === 'MEDIA') return `${dias} días activo — parámetros tomando forma con ${viajes} viajes`
  if (confianza === 'ALTA') return `Sistema calibrado — ${viajes} viajes analizados en ${dias} días`
  return `Sistema experto — modelo altamente preciso con ${viajes} viajes en ${dias} días`
}
